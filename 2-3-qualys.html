Qualys Platform Overview

Qualys is a cloud-native security platform offering multiple integrated applications. Unlike traditional scanners, Qualys is delivered entirely as SaaS with no on-premises management infrastructure required.

Cloud Agent Deployment

Qualys Cloud Agents are lightweight software installed on endpoints that continuously collect data and report to the Qualys platform without network-based scanning.

Agent Installation

Bash 
# Windows Agent Installation (Command Line)
# Download from Qualys portal with your activation key

QualysCloudAgent.exe CustomerId={CUSTOMER_ID} ActivationId={ACTIVATION_ID} WebServiceUri=https://qagpublic.qg2.apps.qualys.com/CloudAgent/

# Silent installation
msiexec /i QualysCloudAgent.msi /quiet CUSTOMID={CUSTOMER_ID} ACTIVATIONID={ACTIVATION_ID} WEBSERVICEURI=https://qagpublic.qg2.apps.qualys.com/CloudAgent/

# Linux Agent Installation
# Download agent package from Qualys portal
curl -o qualys-cloud-agent.x86_64.rpm "https://qualysguard.qualys.com/cloud-agent/..."

# Install RPM (RHEL/CentOS)
sudo rpm -ivh qualys-cloud-agent.x86_64.rpm

# Activate agent
sudo /usr/local/qualys/cloud-agent/bin/qualys-cloud-agent.sh ActivationId={ACTIVATION_ID} CustomerId={CUSTOMER_ID}

# Check agent status
sudo systemctl status qualys-cloud-agent

# Agent logs
tail -f /var/log/qualys/qualys-cloud-agent.log

# Debian/Ubuntu
sudo dpkg -i qualys-cloud-agent_x.x.x.x_amd64.deb
sudo /usr/local/qualys/cloud-agent/bin/qualys-cloud-agent.sh ActivationId={ACTIVATION_ID} CustomerId={CUSTOMER_ID}

"""
Qualys API Integration Module
Provides VMDR automation capabilities.
"""

import requests
import base64
from typing import Dict, List, Optional
import xml.etree.ElementTree as ET


class QualysAPI:
    """
    Qualys API client for vulnerability management.
    
    Supports both VMDR API (REST) and legacy QualysGuard API (XML).
    """
    
    def __init__(self, username: str, password: str, platform: str = 'qg2'):
        """
        Initialize Qualys API client.
        
        Args:
            username: Qualys username
            password: Qualys password
            platform: Qualys platform (qg1, qg2, qg3, qg4)
        """
        self.base_url = f'https://qualysapi.{platform}.apps.qualys.com'
        self.gateway_url = f'https://gateway.{platform}.apps.qualys.com'
        self.auth = base64.b64encode(f'{username}:{password}'.encode()).decode()
        self.headers = {
            'Authorization': f'Basic {self.auth}',
            'X-Requested-With': 'Python'
        }
    
    # â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
    # ASSET MANAGEMENT
    # â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
    
    def search_assets(self, qql_query: str, limit: int = 100) -> List[Dict]:
        """
        Search assets using QQL.
        
        Args:
            qql_query: Qualys Query Language query
            limit: Maximum results to return
            
        Returns:
            List of matching assets
        """
        url = f'{self.gateway_url}/am/v1/assets'
        params = {
            'filter': qql_query,
            'pageSize': limit
        }
        
        response = requests.get(url, headers=self.headers, params=params)
        response.raise_for_status()
        
        return response.json().get('assets', [])
    
    def get_asset_details(self, asset_id: str) -> Dict:
        """Get detailed information for a specific asset."""
        url = f'{self.gateway_url}/am/v1/assets/{asset_id}'
        response = requests.get(url, headers=self.headers)
        response.raise_for_status()
        return response.json()
    
    def add_asset_tags(self, asset_ids: List[str], tag_ids: List[str]) -> bool:
        """Add tags to assets."""
        url = f'{self.gateway_url}/am/v1/assets/tag'
        data = {
            'assetIds': asset_ids,
            'tagIds': tag_ids
        }
        response = requests.post(url, headers=self.headers, json=data)
        return response.status_code == 200
    
    # â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
    # VULNERABILITY MANAGEMENT
    # â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
    
    def get_vulnerabilities(self, qql_query: str, limit: int = 1000) -> List[Dict]:
        """
        Search vulnerabilities using QQL.
        
        Args:
            qql_query: QQL filter (e.g., "vulnerabilities.severity:5")
            limit: Maximum results
            
        Returns:
            List of vulnerability findings
        """
        url = f'{self.gateway_url}/vmdr/v2/vulnerabilities'
        params = {
            'filter': qql_query,
            'pageSize': min(limit, 1000)
        }
        
        response = requests.get(url, headers=self.headers, params=params)
        response.raise_for_status()
        
        return response.json().get('vulnerabilities', [])
    
    def get_vulnerability_count(self, qql_query: str = '') -> Dict:
        """Get vulnerability counts by severity."""
        url = f'{self.gateway_url}/vmdr/v2/vulnerabilities/count'
        params = {'filter': qql_query} if qql_query else {}
        
        response = requests.get(url, headers=self.headers, params=params)
        response.raise_for_status()
        
        return response.json()
    
    def get_kev_vulnerabilities(self) -> List[Dict]:
        """Get CISA KEV (Known Exploited Vulnerabilities)."""
        return self.get_vulnerabilities('vulnerabilities.knownExploited:true')
    
    # â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
    # SCANNING
    # â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
    
    def launch_scan(self, scan_title: str, ip_list: str, 
                    option_id: str, scanner_name: str = None) -> str:
        """
        Launch a vulnerability scan.
        
        Args:
            scan_title: Name for the scan
            ip_list: Comma-separated IPs or ranges
            option_id: Scan option profile ID
            scanner_name: Specific scanner appliance
            
        Returns:
            Scan reference ID
        """
        url = f'{self.base_url}/api/2.0/fo/scan/'
        data = {
            'action': 'launch',
            'scan_title': scan_title,
            'ip': ip_list,
            'option_id': option_id
        }
        
        if scanner_name:
            data['iscanner_name'] = scanner_name
        
        response = requests.post(url, headers=self.headers, data=data)
        response.raise_for_status()
        
        # Parse XML response for scan reference
        root = ET.fromstring(response.text)
        ref = root.find('.//VALUE')
        return ref.text if ref is not None else None
    
    def get_scan_status(self, scan_ref: str) -> str:
        """Get scan status by reference ID."""
        url = f'{self.base_url}/api/2.0/fo/scan/'
        params = {
            'action': 'list',
            'scan_ref': scan_ref
        }
        
        response = requests.get(url, headers=self.headers, params=params)
        root = ET.fromstring(response.text)
        status = root.find('.//STATE')
        return status.text if status is not None else 'Unknown'
    
    # â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
    # REPORTING
    # â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
    
    def generate_report(self, report_type: str, template_id: str,
                        qql_filter: str = None) -> str:
        """
        Generate a report.
        
        Args:
            report_type: 'Scan', 'Remediation', 'Compliance'
            template_id: Report template ID
            qql_filter: Optional QQL filter
            
        Returns:
            Report ID
        """
        url = f'{self.base_url}/api/2.0/fo/report/'
        data = {
            'action': 'launch',
            'report_type': report_type,
            'template_id': template_id,
            'output_format': 'pdf'
        }
        
        if qql_filter:
            data['filter'] = qql_filter
        
        response = requests.post(url, headers=self.headers, data=data)
        response.raise_for_status()
        
        root = ET.fromstring(response.text)
        report_id = root.find('.//VALUE')
        return report_id.text if report_id is not None else None
    
    def get_trurisk_score(self) -> Dict:
        """Get organization TruRisk score summary."""
        url = f'{self.gateway_url}/vmdr/v2/trurisk/score'
        response = requests.get(url, headers=self.headers)
        response.raise_for_status()
        return response.json()
