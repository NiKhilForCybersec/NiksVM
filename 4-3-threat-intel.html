CISA KEV (Known Exploited Vulnerabilities)

The CISA KEV catalog contains vulnerabilities confirmed to be actively exploited in the wild. It's the most authoritative source for exploitation status.

Python 
"""
CISA KEV Integration
Download and query the Known Exploited Vulnerabilities catalog.
"""

import requests
from typing import Dict, List, Optional, Set
from datetime import datetime
import json


class CISAKEVClient:
    """Client for CISA Known Exploited Vulnerabilities catalog."""
    
    KEV_URL = "https://www.cisa.gov/sites/default/files/feeds/known_exploited_vulnerabilities.json"
    
    def __init__(self):
        self.vulnerabilities = {}
        self.cve_set = set()
        self.last_updated = None
    
    def refresh(self) -> int:
        """Download and parse KEV catalog."""
        response = requests.get(self.KEV_URL)
        response.raise_for_status()
        
        data = response.json()
        self.last_updated = data.get('catalogVersion')
        
        self.vulnerabilities = {}
        for vuln in data.get('vulnerabilities', []):
            cve_id = vuln.get('cveID')
            self.vulnerabilities[cve_id] = {
                'cve_id': cve_id,
                'vendor': vuln.get('vendorProject'),
                'product': vuln.get('product'),
                'name': vuln.get('vulnerabilityName'),
                'description': vuln.get('shortDescription'),
                'date_added': vuln.get('dateAdded'),
                'due_date': vuln.get('dueDate'),
                'required_action': vuln.get('requiredAction'),
                'notes': vuln.get('notes'),
                'known_ransomware': vuln.get('knownRansomwareCampaignUse', 'Unknown')
            }
        
        self.cve_set = set(self.vulnerabilities.keys())
        return len(self.vulnerabilities)
    
    def is_kev(self, cve_id: str) -> bool:
        """Check if CVE is in KEV catalog."""
        return cve_id.upper() in self.cve_set
    
    def get_details(self, cve_id: str) -> Optional[Dict]:
        """Get KEV details for a CVE."""
        return self.vulnerabilities.get(cve_id.upper())
    
    def check_batch(self, cve_ids: List[str]) -> Dict[str, bool]:
        """Check multiple CVEs against KEV."""
        return {cve: self.is_kev(cve) for cve in cve_ids}
    
    def get_ransomware_vulns(self) -> List[Dict]:
        """Get vulnerabilities known to be used in ransomware."""
        return [
            v for v in self.vulnerabilities.values() 
            if v.get('known_ransomware') == 'Known'
        ]
    
    def get_recent_additions(self, days: int = 30) -> List[Dict]:
        """Get recently added KEV entries."""
        cutoff = datetime.now().date()
        recent = []
        
        for vuln in self.vulnerabilities.values():
            try:
                added_date = datetime.strptime(
                    vuln['date_added'], '%Y-%m-%d'
                ).date()
                delta = (cutoff - added_date).days
                if delta <= days:
                    recent.append(vuln)
            except (ValueError, KeyError):
                pass
        
        return sorted(recent, key=lambda x: x['date_added'], reverse=True)


# Example usage
kev = CISAKEVClient()
count = kev.refresh()
print(f"Loaded {count} KEV vulnerabilities")

# Check specific CVEs
vulns_to_check = ['CVE-2021-44228', 'CVE-2023-1234', 'CVE-2024-0000']
results = kev.check_batch(vulns_to_check)

for cve, is_kev in results.items():
    if is_kev:
        details = kev.get_details(cve)
        print(f"âš ï¸  {cve} is in KEV - Due: {details['due_date']}")
    else:
        print(f"âœ“  {cve} is not in KEV")

# Ransomware-related vulns
ransomware = kev.get_ransomware_vulns()
print(f"\n{len(ransomware)} vulnerabilities linked to ransomware")

class NVDClient:
    """Query NVD for vulnerability details and references."""
    
    NVD_API = "https://services.nvd.nist.gov/rest/json/cves/2.0"
    
    def __init__(self, api_key: str = None):
        self.api_key = api_key
        self.headers = {'apiKey': api_key} if api_key else {}
    
    def get_cve(self, cve_id: str) -> Optional[Dict]:
        """Get CVE details from NVD."""
        response = requests.get(
            self.NVD_API,
            params={'cveId': cve_id},
            headers=self.headers
        )
        response.raise_for_status()
        
        data = response.json()
        vulns = data.get('vulnerabilities', [])
        return vulns[0] if vulns else None
    
    def has_exploit_reference(self, cve_id: str) -> Dict:
        """Check if CVE has exploit references in NVD."""
        cve_data = self.get_cve(cve_id)
        if not cve_data:
            return {'has_exploit': False, 'sources': []}
        
        exploit_sources = []
        cve_item = cve_data.get('cve', {})
        
        for ref in cve_item.get('references', []):
            tags = ref.get('tags', [])
            if 'Exploit' in tags:
                exploit_sources.append({
                    'url': ref.get('url'),
                    'source': ref.get('source')
                })
        
        return {
            'has_exploit': len(exploit_sources) > 0,
            'sources': exploit_sources
        }


class VulnCheckClient:
    """VulnCheck API for exploit intelligence."""
    
    API_URL = "https://api.vulncheck.com/v3"
    
    def __init__(self, api_key: str):
        self.headers = {'Authorization': f'Bearer {api_key}'}
    
    def get_exploit_status(self, cve_id: str) -> Dict:
        """Get exploit status from VulnCheck."""
        response = requests.get(
            f"{self.API_URL}/cve/{cve_id}",
            headers=self.headers
        )
        response.raise_for_status()
        
        data = response.json()
        return {
            'has_poc': data.get('has_poc', False),
            'has_exploit': data.get('has_exploit', False),
            'in_the_wild': data.get('exploited_in_wild', False),
            'exploit_maturity': data.get('exploit_maturity', 'unknown')
        }


class ExploitEnricher:
    """Aggregate exploit data from multiple sources."""
    
    def __init__(self, nvd_key: str = None, vulncheck_key: str = None):
        self.nvd = NVDClient(nvd_key) if nvd_key else None
        self.vulncheck = VulnCheckClient(vulncheck_key) if vulncheck_key else None
        self.kev = CISAKEVClient()
        self.kev.refresh()
    
    def enrich(self, cve_id: str) -> Dict:
        """Enrich CVE with exploit intelligence."""
        enrichment = {
            'cve_id': cve_id,
            'cisa_kev': self.kev.is_kev(cve_id),
            'exploit_available': False,
            'exploit_maturity': 'unknown',
            'sources': []
        }
        
        # Check CISA KEV
        if enrichment['cisa_kev']:
            enrichment['exploit_available'] = True
            enrichment['exploit_maturity'] = 'in_the_wild'
            kev_details = self.kev.get_details(cve_id)
            enrichment['kev_due_date'] = kev_details.get('due_date')
            enrichment['known_ransomware'] = kev_details.get('known_ransomware')
        
        # Check NVD references
        if self.nvd:
            nvd_result = self.nvd.has_exploit_reference(cve_id)
            if nvd_result['has_exploit']:
                enrichment['exploit_available'] = True
                enrichment['sources'].extend(nvd_result['sources'])
        
        # Check VulnCheck
        if self.vulncheck:
            vc_result = self.vulncheck.get_exploit_status(cve_id)
            if vc_result['has_exploit']:
                enrichment['exploit_available'] = True
            if vc_result['in_the_wild']:
                enrichment['exploit_maturity'] = 'in_the_wild'
            elif vc_result['has_exploit']:
                enrichment['exploit_maturity'] = 'functional'
            elif vc_result['has_poc']:
                enrichment['exploit_maturity'] = 'poc'
        
        return enrichment
