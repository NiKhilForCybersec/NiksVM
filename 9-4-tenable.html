def export_critical_vulns():
    """Export critical vulnerabilities from last 30 days."""
    
    vulns = tio.exports.vulns(
        severity=['critical', 'high'],
        state=['open', 'reopened'],
        since=int((datetime.now() - timedelta(days=30)).timestamp())
    )
    
    for vuln in vulns:
        print(f"Plugin: {vuln['plugin']['id']}")
        print(f"CVE: {vuln['plugin'].get('cve', ['N/A'])}")
        print(f"VPR: {vuln['plugin'].get('vpr', {}).get('score', 'N/A')}")
        print(f"Asset: {vuln['asset']['hostname']}")
        print("---")

def launch_remediation_scan(scan_id: int, targets: list = None):
    """Launch a scan for remediation verification."""
    
    if targets:
        # Launch with specific targets
        response = tio.scans.launch(scan_id, targets=targets)
    else:
        # Launch full scan
        response = tio.scans.launch(scan_id)
    
    scan_uuid = response['scan_uuid']
    print(f"Scan launched: {scan_uuid}")
    
    return scan_uuid

"How does Tenable's VPR differ from CVSS?"

Sample answer: "CVSS is a static score based on the vulnerability's technical characteristicsâ€”how it's exploited, what impact it has. VPR (Vulnerability Priority Rating) is dynamic and incorporates threat intelligence: Is there exploit code available? Is it being discussed on dark web forums? Has it been seen in the wild? A vulnerability might have CVSS 9.0 but if there's no public exploit, VPR might be lower. Conversely, a CVSS 7.0 vulnerability with active Metasploit module gets higher VPR. I use VPR to prioritize what to patch first."

Summary

Tenable's VPR scoring provides risk-based prioritization beyond CVSS. Use the exports API for bulk data extraction, agents for endpoints and cloud workloads, and network scans for infrastructure. Combine VPR with asset criticality for effective prioritization.
