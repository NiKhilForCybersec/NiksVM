Metric Calculations

SQL 
-- MTTR by Severity
SELECT 
    severity,
    AVG(DATEDIFF(day, first_detected, remediated_date)) as mttr_days,
    COUNT(*) as vuln_count
FROM vulnerabilities
WHERE remediated_date IS NOT NULL
  AND remediated_date >= DATEADD(month, -1, GETDATE())
GROUP BY severity;

-- SLA Compliance
SELECT 
    severity,
    COUNT(*) as total,
    SUM(CASE WHEN remediated_date <= sla_due_date THEN 1 ELSE 0 END) as within_sla,
    ROUND(SUM(CASE WHEN remediated_date <= sla_due_date THEN 1 ELSE 0 END) * 100.0 / COUNT(*), 1) as compliance_pct
FROM vulnerabilities
WHERE remediated_date IS NOT NULL
  AND remediated_date >= DATEADD(month, -1, GETDATE())
GROUP BY severity;

-- Vulnerability Age Distribution
SELECT 
    CASE 
        WHEN DATEDIFF(day, first_detected, GETDATE()) <= 30 THEN '0-30 days'
        WHEN DATEDIFF(day, first_detected, GETDATE()) <= 60 THEN '31-60 days'
        WHEN DATEDIFF(day, first_detected, GETDATE()) <= 90 THEN '61-90 days'
        ELSE '90+ days'
    END as age_bucket,
    COUNT(*) as vuln_count
FROM vulnerabilities
WHERE status = 'Open'
GROUP BY 
    CASE 
        WHEN DATEDIFF(day, first_detected, GETDATE()) <= 30 THEN '0-30 days'
        WHEN DATEDIFF(day, first_detected, GETDATE()) <= 60 THEN '31-60 days'
        WHEN DATEDIFF(day, first_detected, GETDATE()) <= 90 THEN '61-90 days'
        ELSE '90+ days'
    END
ORDER BY 1;
