Kubernetes Security

YAML 
# Secure Pod Configuration Example
apiVersion: v1
kind: Pod
metadata:
  name: secure-pod
spec:
  # Don't run as root
  securityContext:
    runAsNonRoot: true
    runAsUser: 1000
    fsGroup: 1000
    
  containers:
  - name: app
    image: myapp:v1.0
    
    # Container-level security
    securityContext:
      allowPrivilegeEscalation: false
      readOnlyRootFilesystem: true
      capabilities:
        drop:
          - ALL
    
    # Resource limits (prevent DoS)
    resources:
      limits:
        memory: "256Mi"
        cpu: "500m"
      requests:
        memory: "128Mi"
        cpu: "250m"
    
    # Health checks
    livenessProbe:
      httpGet:
        path: /health
        port: 8080
    
    # Don't mount service account token if not needed
  automountServiceAccountToken: false
