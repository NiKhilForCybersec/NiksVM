"""
ServiceNow Vulnerability Integration
Create and manage vulnerability items and incidents.
"""

import requests
from typing import Dict, List, Optional
from datetime import datetime, timedelta


class ServiceNowVulnClient:
    """ServiceNow integration for vulnerability management."""
    
    def __init__(self, instance: str, username: str, password: str):
        self.base_url = f"https://{instance}.service-now.com/api/now"
        self.auth = (username, password)
        self.headers = {
            'Content-Type': 'application/json',
            'Accept': 'application/json'
        }
    
    def create_vulnerable_item(self, vuln: Dict, asset: Dict) -> str:
        """
        Create Vulnerable Item in ServiceNow VR module.
        
        Uses the sn_vul_vulnerable_item table for native VM integration.
        """
        payload = {
            'vulnerability': vuln.get('cve_id'),
            'configuration_item': asset.get('cmdb_sys_id'),
            'source': 'Tenable',  # or Qualys, Rapid7, etc.
            'severity': self._map_severity(vuln.get('cvss_score', 5.0)),
            'cvss_score': vuln.get('cvss_score'),
            'first_found': vuln.get('first_seen'),
            'last_found': vuln.get('last_seen'),
            'state': 'Open',
            'risk_score': vuln.get('risk_score'),
            'solution': vuln.get('solution', 'Apply vendor patch')
        }
        
        response = requests.post(
            f"{self.base_url}/table/sn_vul_vulnerable_item",
            auth=self.auth,
            headers=self.headers,
            json=payload
        )
        response.raise_for_status()
        
        return response.json()['result']['sys_id']
    
    def create_security_incident(self, vuln: Dict, assets: List[Dict]) -> str:
        """Create security incident for critical vulnerabilities."""
        
        asset_list = '\n'.join(f"- {a.get('hostname')}" for a in assets[:20])
        
        payload = {
            'short_description': f"Critical Vulnerability: {vuln.get('cve_id')}",
            'description': f"""
Critical vulnerability requiring immediate attention.

CVE: {vuln.get('cve_id')}
CVSS Score: {vuln.get('cvss_score')}
CISA KEV: {'Yes' if vuln.get('cisa_kev') else 'No'}

Affected Assets ({len(assets)}):
{asset_list}

Solution: {vuln.get('solution', 'Apply vendor patch')}
            """,
            'category': 'Security',
            'subcategory': 'Vulnerability',
            'impact': '2',  # Medium-High
            'urgency': '1' if vuln.get('cisa_kev') else '2',
            'priority': self._calculate_snow_priority(vuln),
            'assignment_group': 'Security Operations',
            'u_cve_id': vuln.get('cve_id'),
            'u_affected_asset_count': len(assets)
        }
        
        response = requests.post(
            f"{self.base_url}/table/sn_si_incident",
            auth=self.auth,
            headers=self.headers,
            json=payload
        )
        response.raise_for_status()
        
        return response.json()['result']['number']
    
    def sync_remediation_status(self, vuln_item_id: str, 
                                 scanner_status: str) -> bool:
        """Sync status from scanner back to ServiceNow."""
        
        status_map = {
            'remediated': 'Resolved',
            'fixed': 'Resolved',
            'open': 'Open',
            'in_progress': 'Work in Progress',
            'accepted': 'Risk Accepted'
        }
        
        snow_state = status_map.get(scanner_status.lower(), 'Open')
        
        payload = {
            'state': snow_state,
            'work_notes': f"Status synced from scanner: {scanner_status}"
        }
        
        if snow_state == 'Resolved':
            payload['resolved_at'] = datetime.now().isoformat()
        
        response = requests.patch(
            f"{self.base_url}/table/sn_vul_vulnerable_item/{vuln_item_id}",
            auth=self.auth,
            headers=self.headers,
            json=payload
        )
        
        return response.status_code == 200
    
    def _map_severity(self, cvss: float) -> str:
        """Map CVSS to ServiceNow severity."""
        if cvss >= 9.0:
            return '1'  # Critical
        elif cvss >= 7.0:
            return '2'  # High
        elif cvss >= 4.0:
            return '3'  # Medium
        return '4'  # Low
    
    def _calculate_snow_priority(self, vuln: Dict) -> str:
        """Calculate ServiceNow priority."""
        if vuln.get('cisa_kev') or vuln.get('cvss_score', 0) >= 9.5:
            return '1'  # Critical
        elif vuln.get('cvss_score', 0) >= 8.0:
            return '2'  # High
        elif vuln.get('cvss_score', 0) >= 6.0:
            return '3'  # Moderate
        return '4'  # Low

"""
Jira Vulnerability Ticket Integration
Create and manage vulnerability tickets in Jira.
"""

from jira import JIRA
from typing import Dict, List, Optional


class JiraVulnClient:
    """Jira integration for vulnerability management."""
    
    def __init__(self, server: str, username: str, api_token: str,
                 project_key: str = 'SEC'):
        self.jira = JIRA(
            server=server,
            basic_auth=(username, api_token)
        )
        self.project_key = project_key
    
    def create_vuln_ticket(self, vuln: Dict, assets: List[Dict],
                           priority: str = 'P2') -> str:
        """
        Create vulnerability remediation ticket.
        
        Args:
            vuln: Vulnerability details
            assets: List of affected assets
            priority: Priority level (P1-P4)
        
        Returns:
            Jira issue key
        """
        # Build asset table
        asset_table = "||Hostname||IP||Criticality||\n"
        for asset in assets[:20]:
            asset_table += f"|{asset.get('hostname')}|{asset.get('ip')}|Tier {asset.get('criticality')}|\n"
        
        if len(assets) > 20:
            asset_table += f"|... and {len(assets) - 20} more|||\n"
        
        description = f"""
h2. Vulnerability Details

*CVE:* {vuln.get('cve_id')}
*CVSS Score:* {vuln.get('cvss_score')}
*EPSS:* {vuln.get('epss', 'N/A')}
*CISA KEV:* {'Yes âš ï¸' if vuln.get('cisa_kev') else 'No'}

h3. Description
{vuln.get('description', 'See vendor advisory for details.')}

h3. Solution
{vuln.get('solution', 'Apply vendor-provided patch.')}

h2. Affected Assets ({len(assets)} total)

{asset_table}

h2. Remediation Steps
# Review patch documentation
# Test in non-production environment
# Schedule maintenance window
# Apply remediation
# Verify system functionality
# Mark ticket for verification rescan
        """
        
        # Map priority
        jira_priority = {
            'P1': 'Highest',
            'P2': 'High',
            'P3': 'Medium',
            'P4': 'Low'
        }.get(priority, 'Medium')
        
        issue_dict = {
            'project': {'key': self.project_key},
            'summary': f"[{priority}] Remediate {vuln.get('cve_id')} - {len(assets)} assets",
            'description': description,
            'issuetype': {'name': 'Task'},
            'priority': {'name': jira_priority},
            'labels': ['vulnerability', 'security', vuln.get('cve_id', 'unknown')],
        }
        
        # Add custom fields if configured
        if vuln.get('cve_id'):
            issue_dict['customfield_10001'] = vuln.get('cve_id')  # CVE field
        if vuln.get('cvss_score'):
            issue_dict['customfield_10002'] = vuln.get('cvss_score')  # CVSS field
        
        issue = self.jira.create_issue(fields=issue_dict)
        
        return issue.key
    
    def add_assets_as_subtasks(self, parent_key: str, assets: List[Dict]):
        """Create subtasks for each affected asset."""
        
        for asset in assets:
            subtask = {
                'project': {'key': self.project_key},
                'summary': f"Patch {asset.get('hostname')}",
                'description': f"Apply remediation to {asset.get('hostname')} ({asset.get('ip')})",
                'issuetype': {'name': 'Sub-task'},
                'parent': {'key': parent_key}
            }
            self.jira.create_issue(fields=subtask)
    
    def update_ticket_status(self, issue_key: str, status: str, 
                              comment: str = None):
        """Update ticket status based on scanner findings."""
        
        # Transition mapping (IDs vary by Jira configuration)
        transitions = {
            'in_progress': '21',
            'remediated': '31',
            'verified': '41',
            'closed': '51'
        }
        
        transition_id = transitions.get(status.lower())
        if transition_id:
            self.jira.transition_issue(issue_key, transition_id)
        
        if comment:
            self.jira.add_comment(issue_key, comment)
    
    def link_related_tickets(self, issue_key: str, related_cves: List[str]):
        """Link tickets for related vulnerabilities."""
        
        # Search for existing tickets with related CVEs
        for cve in related_cves:
            jql = f'project = {self.project_key} AND labels = "{cve}"'
            issues = self.jira.search_issues(jql)
            
            for related_issue in issues:
                if related_issue.key != issue_key:
                    self.jira.create_issue_link(
                        type='Relates',
                        inwardIssue=issue_key,
                        outwardIssue=related_issue.key
                    )

ğŸ’¡ Interview Question

"How do you integrate vulnerability scanners with ticketing systems?"

Sample answer: "I build bidirectional integrations using APIs. When the scanner detects new vulnerabilities, tickets are auto-created in ServiceNow or Jira with CVE details, affected assets, and calculated priority. I aggregate by CVE and owner to avoid ticket floods. When the scanner shows a vulnerability is fixed, tickets auto-update to 'Verified' status. Going the other way, risk acceptances in the ticketing system sync back to create scanner exceptions. I use webhooks or scheduled syncs depending on the platform capabilities."

Summary

Ticketing integration connects vulnerability findings to remediation workflows. Build bidirectional sync so tickets reflect scanner status and vice versa. Aggregate intelligently to avoid ticket floods while maintaining accountability.
