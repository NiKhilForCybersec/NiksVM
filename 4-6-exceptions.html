-- Exception Tracking Database Schema

CREATE TABLE vulnerability_exceptions (
    exception_id        UUID PRIMARY KEY,
    
    -- Vulnerability Reference
    vulnerability_id    UUID NOT NULL,
    cve_id              VARCHAR(20),
    asset_id            UUID NOT NULL,
    cvss_score          DECIMAL(3,1),
    risk_level          VARCHAR(20),  -- Critical, High, Medium, Low
    
    -- Exception Details
    exception_type      VARCHAR(50),  -- Technical, Business, Risk Acceptance
    justification       TEXT NOT NULL,
    business_impact     TEXT,
    
    -- Compensating Controls
    compensating_controls TEXT,
    controls_validated   BOOLEAN DEFAULT FALSE,
    validation_date      DATE,
    validated_by         VARCHAR(100),
    
    -- Approval
    approval_status     VARCHAR(20) DEFAULT 'Pending',
    approved_by         VARCHAR(100),
    approval_date       DATE,
    approval_comments   TEXT,
    
    -- Timeline
    requested_date      DATE NOT NULL DEFAULT CURRENT_DATE,
    start_date          DATE,
    expiration_date     DATE NOT NULL,
    remediation_plan    TEXT,
    remediation_date    DATE,  -- Planned remediation
    
    -- Tracking
    requestor           VARCHAR(100) NOT NULL,
    business_owner      VARCHAR(100) NOT NULL,
    review_frequency    VARCHAR(20) DEFAULT 'Monthly',
    last_review_date    DATE,
    next_review_date    DATE,
    
    -- Status
    status              VARCHAR(20) DEFAULT 'Active',
    closure_reason      TEXT,
    closed_date         DATE
);

-- Exception review tracking
CREATE TABLE exception_reviews (
    review_id       UUID PRIMARY KEY,
    exception_id    UUID REFERENCES vulnerability_exceptions(exception_id),
    review_date     DATE NOT NULL,
    reviewer        VARCHAR(100),
    status_update   TEXT,
    controls_effective BOOLEAN,
    recommendation  VARCHAR(50),  -- Renew, Close, Escalate
    notes           TEXT
);

-- Useful queries

-- Exceptions expiring in next 30 days
SELECT e.*, v.cve_id, a.hostname
FROM vulnerability_exceptions e
JOIN vulnerabilities v ON e.vulnerability_id = v.id
JOIN assets a ON e.asset_id = a.id
WHERE e.status = 'Active'
  AND e.expiration_date BETWEEN CURRENT_DATE AND CURRENT_DATE + 30;

-- Exception metrics by status
SELECT 
    risk_level,
    COUNT(*) as total,
    COUNT(*) FILTER (WHERE status = 'Active') as active,
    COUNT(*) FILTER (WHERE expiration_date < CURRENT_DATE) as expired,
    AVG(expiration_date - start_date) as avg_duration_days
FROM vulnerability_exceptions
GROUP BY risk_level
ORDER BY 
    CASE risk_level 
        WHEN 'Critical' THEN 1 
        WHEN 'High' THEN 2 
        WHEN 'Medium' THEN 3 
        ELSE 4 
    END;
âœ… Exception Governance

Every exception should have: (1) a documented business justification, (2) compensating controls that are validated, (3) an expiration date with no indefinite exceptions, (4) a remediation plan, and (5) regular review. Track exception metrics to identify systemic issuesâ€”high exception rates may indicate resource problems or unrealistic SLAs.

"How do you handle vulnerability exceptions and risk acceptance?"

Sample answer: "Exceptions are sometimes necessary but need strong governance. I require a formal request with business justification, proposed compensating controls, and a remediation timeline. Critical vulnerabilities require CISO approval. I never allow indefinite exceptionsâ€”maximum 90 days with renewal. Compensating controls must be validated and monitored. I track exception metrics to identify patternsâ€”if one team has excessive exceptions, that indicates a resource or process problem to address. Summary

Exception management balances security with business reality. Not every vulnerability can be patched immediately, but exceptions must be governed: documented justification, compensating controls, finite duration, and appropriate approval authority. Track exception metrics as a program health indicator.
