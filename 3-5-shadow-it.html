Shadow IT Detection

ğŸ“– Section 3.5 â±ï¸ 14 min read

This page covers detecting shadow IT including unauthorized infrastructure, rogue cloud accounts, unsanctioned SaaS applications, and strategies for bringing shadow IT under proper governance and security controls.

What is Shadow IT?

Shadow IT refers to technology deployed without IT department approval or oversight. These assets are particularly dangerous because they typically lack security controls, patching, and monitoring.

Network-Based Detection

"""
Shadow IT Detection - Network Analysis
Compare discovered assets against known inventory.
"""

from typing import Dict, List, Set
from datetime import datetime


class ShadowITDetector:
    """Identify potential shadow IT through inventory comparison."""
    
    def __init__(self, authorized_sources: List[str]):
        """
        Args:
            authorized_sources: List of authoritative inventory sources
                               e.g., ['cmdb', 'ad', 'intune']
        """
        self.authorized_sources = authorized_sources
        self.shadow_assets = []
    
    def detect_shadow_assets(self, 
                              discovered_assets: List[Dict],
                              authorized_assets: List[Dict]) -> List[Dict]:
        """
        Find assets discovered by scanners but not in authorized inventory.
        
        Args:
            discovered_assets: Assets from network/vuln scanning
            authorized_assets: Assets from CMDB, AD, etc.
        
        Returns:
            List of potential shadow IT assets
        """
        # Build sets of authorized identifiers
        authorized_ips = {a.get('ip') for a in authorized_assets if a.get('ip')}
        authorized_macs = {a.get('mac', '').lower() for a in authorized_assets if a.get('mac')}
        authorized_hosts = {a.get('hostname', '').lower() for a in authorized_assets if a.get('hostname')}
        
        shadow_assets = []
        
        for asset in discovered_assets:
            is_authorized = False
            
            # Check against all identifiers
            if asset.get('ip') in authorized_ips:
                is_authorized = True
            elif asset.get('mac', '').lower() in authorized_macs:
                is_authorized = True
            elif asset.get('hostname', '').lower() in authorized_hosts:
                is_authorized = True
            
            if not is_authorized:
                shadow_asset = asset.copy()
                shadow_asset['shadow_detected_at'] = datetime.utcnow().isoformat()
                shadow_asset['shadow_reason'] = 'Not found in authorized inventory'
                shadow_assets.append(shadow_asset)
        
        self.shadow_assets = shadow_assets
        return shadow_assets
    
    def classify_shadow_risk(self, asset: Dict) -> str:
        """Classify risk level of shadow asset."""
        risk_score = 0
        
        # Internet-facing is high risk
        if asset.get('public_ip') or asset.get('internet_facing'):
            risk_score += 40
        
        # Running services increase risk
        open_ports = asset.get('open_ports', [])
        if len(open_ports) > 10:
            risk_score += 20
        elif len(open_ports) > 5:
            risk_score += 10
        
        # Server OS vs workstation
        os_name = asset.get('os', '').lower()
        if 'server' in os_name:
            risk_score += 20
        
        # Has vulnerabilities
        vuln_count = asset.get('vulnerability_count', 0)
        if vuln_count > 10:
            risk_score += 30
        elif vuln_count > 0:
            risk_score += 15
        
        # Classify
        if risk_score >= 60:
            return 'Critical'
        elif risk_score >= 40:
            return 'High'
        elif risk_score >= 20:
            return 'Medium'
        else:
            return 'Low'
    
    def generate_report(self) -> Dict:
        """Generate shadow IT summary report."""
        report = {
            'generated_at': datetime.utcnow().isoformat(),
            'total_shadow_assets': len(self.shadow_assets),
            'by_risk': {'Critical': 0, 'High': 0, 'Medium': 0, 'Low': 0},
            'by_os': {},
            'assets': []
        }
        
        for asset in self.shadow_assets:
            risk = self.classify_shadow_risk(asset)
            report['by_risk'][risk] += 1
            
            os_type = asset.get('os_family', 'Unknown')
            report['by_os'][os_type] = report['by_os'].get(os_type, 0) + 1
            
            report['assets'].append({
                'ip': asset.get('ip'),
                'hostname': asset.get('hostname'),
                'os': asset.get('os'),
                'risk_level': risk,
                'open_ports': len(asset.get('open_ports', [])),
                'first_seen': asset.get('first_discovered')
            })
        
        return report


# Example usage
detector = ShadowITDetector(['cmdb', 'ad'])

# Assets from network scanning
discovered = [
    {'ip': '192.168.1.10', 'hostname': 'web-01', 'os': 'Linux'},
    {'ip': '192.168.1.99', 'hostname': 'dev-test-01', 'os': 'Windows Server 2019',
     'open_ports': [22, 80, 443, 3389]},
    {'ip': '192.168.1.100', 'hostname': 'raspberry-pi', 'os': 'Linux',
     'open_ports': [22, 8080]}
]

# Assets from CMDB
authorized = [
    {'ip': '192.168.1.10', 'hostname': 'web-01', 'os': 'Linux'}
]

shadow = detector.detect_shadow_assets(discovered, authorized)
print(f"Found {len(shadow)} shadow IT assets")
