Automated Verification

"""
Automated Remediation Verification
Trigger rescans and validate fixes.
"""

from typing import Dict, List
from datetime import datetime, timedelta
import time


class RemediationVerifier:
    """Verify vulnerability remediation through rescanning."""
    
    def __init__(self, scanner_client, ticketing_client):
        self.scanner = scanner_client
        self.tickets = ticketing_client
    
    def verify_remediation(self, ticket_id: str) -> Dict:
        """
        Verify remediation for a ticket.
        
        Args:
            ticket_id: Remediation ticket ID
        
        Returns:
            Verification result
        """
        # Get ticket details
        ticket = self.tickets.get_ticket(ticket_id)
        
        cve_id = ticket.get('cve_id')
        assets = ticket.get('affected_assets', [])
        
        results = {
            'ticket_id': ticket_id,
            'cve_id': cve_id,
            'verified_at': datetime.now().isoformat(),
            'assets_verified': 0,
            'assets_still_vulnerable': 0,
            'verification_status': None,
            'details': []
        }
        
        for asset in assets:
            # Trigger targeted rescan
            scan_id = self.scanner.launch_targeted_scan(
                target=asset.get('ip'),
                plugins=[cve_id]  # Scan only for specific vuln
            )
            
            # Wait for scan completion
            self.scanner.wait_for_scan(scan_id, timeout=300)
            
            # Check if vulnerability still present
            findings = self.scanner.get_scan_findings(scan_id, cve_filter=cve_id)
            
            if findings:
                results['assets_still_vulnerable'] += 1
                results['details'].append({
                    'asset': asset.get('hostname'),
                    'status': 'still_vulnerable',
                    'finding': findings[0]
                })
            else:
                results['assets_verified'] += 1
                results['details'].append({
                    'asset': asset.get('hostname'),
                    'status': 'remediated'
                })
        
        # Determine overall status
        if results['assets_still_vulnerable'] == 0:
            results['verification_status'] = 'verified'
            self._close_ticket(ticket_id, results)
        elif results['assets_verified'] > 0:
            results['verification_status'] = 'partial'
            self._update_ticket_partial(ticket_id, results)
        else:
            results['verification_status'] = 'failed'
            self._reopen_ticket(ticket_id, results)
        
        return results
    
    def _close_ticket(self, ticket_id: str, results: Dict):
        """Close ticket after successful verification."""
        self.tickets.update_ticket(ticket_id, {
            'status': 'Closed',
            'resolution': 'Verified Remediated',
            'resolution_notes': f"Verification scan confirmed remediation. "
                               f"{results['assets_verified']} assets verified.",
            'closed_date': datetime.now().isoformat()
        })
    
    def _reopen_ticket(self, ticket_id: str, results: Dict):
        """Reopen ticket if verification fails."""
        self.tickets.update_ticket(ticket_id, {
            'status': 'Reopened',
            'notes': f"Verification FAILED. {results['assets_still_vulnerable']} "
                    f"assets still vulnerable. Please re-attempt remediation."
        })
        
        # Notify assignee
        self.tickets.add_comment(ticket_id, 
            "âš ï¸ Remediation verification failed. Vulnerability still detected. "
            "Please investigate and re-attempt remediation."
        )
    
    def _update_ticket_partial(self, ticket_id: str, results: Dict):
        """Update ticket for partial remediation."""
        self.tickets.add_comment(ticket_id,
            f"Partial verification: {results['assets_verified']} assets remediated, "
            f"{results['assets_still_vulnerable']} still vulnerable."
        )
        
        # Update affected assets list
        still_vulnerable = [
            d['asset'] for d in results['details'] 
            if d['status'] == 'still_vulnerable'
        ]
        
        self.tickets.update_ticket(ticket_id, {
            'affected_assets': still_vulnerable,
            'notes': 'Updated to reflect remaining vulnerable assets'
        })

"How do you verify that vulnerabilities are actually remediated?"

Sample answer: "Tickets don't close based on saying 'we patched it.' I require verification through rescanning. When a ticket is marked remediated, I trigger a targeted scan of the affected assets checking for that specific vulnerability. Only when the scanner confirms the vuln is no longer present does the ticket auto-close. If the vulnerability is still detected, the ticket reopens automatically with notification to the assignee. This prevents premature closure and ensures real remediation."
