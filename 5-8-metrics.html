-- Remediation Metrics Queries

-- MTTR by Severity (last 90 days)
SELECT 
    severity,
    COUNT(*) as remediated_count,
    ROUND(AVG(remediated_date - discovered_date), 1) as mttr_days,
    ROUND(PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY remediated_date - discovered_date), 1) as median_days,
    MIN(remediated_date - discovered_date) as min_days,
    MAX(remediated_date - discovered_date) as max_days
FROM vulnerabilities
WHERE remediated_date IS NOT NULL
  AND remediated_date >= CURRENT_DATE - 90
GROUP BY severity
ORDER BY 
    CASE severity 
        WHEN 'Critical' THEN 1 
        WHEN 'High' THEN 2 
        WHEN 'Medium' THEN 3 
        ELSE 4 
    END;

-- SLA Compliance Rate
SELECT 
    severity,
    COUNT(*) as total_remediated,
    COUNT(*) FILTER (WHERE remediated_date <= sla_due_date) as within_sla,
    ROUND(100.0 * COUNT(*) FILTER (WHERE remediated_date <= sla_due_date) / COUNT(*), 1) as compliance_pct
FROM vulnerabilities
WHERE remediated_date IS NOT NULL
  AND remediated_date >= CURRENT_DATE - 90
GROUP BY severity;

-- Remediation Velocity (weekly trend)
SELECT 
    DATE_TRUNC('week', remediated_date) as week,
    COUNT(*) as closed,
    (SELECT COUNT(*) FROM vulnerabilities v2 
     WHERE DATE_TRUNC('week', v2.discovered_date) = DATE_TRUNC('week', v.remediated_date)) as discovered,
    COUNT(*) - (SELECT COUNT(*) FROM vulnerabilities v2 
                WHERE DATE_TRUNC('week', v2.discovered_date) = DATE_TRUNC('week', v.remediated_date)) as net_change
FROM vulnerabilities v
WHERE remediated_date >= CURRENT_DATE - 90
GROUP BY DATE_TRUNC('week', remediated_date)
ORDER BY week;

-- Backlog Age Distribution
SELECT 
    CASE 
        WHEN CURRENT_DATE - discovered_date <= 30 THEN '0-30 days'
        WHEN CURRENT_DATE - discovered_date <= 60 THEN '31-60 days'
        WHEN CURRENT_DATE - discovered_date <= 90 THEN '61-90 days'
        WHEN CURRENT_DATE - discovered_date <= 180 THEN '91-180 days'
        ELSE '180+ days'
    END as age_bucket,
    severity,
    COUNT(*) as count
FROM vulnerabilities
WHERE remediated_date IS NULL
  AND status NOT IN ('Exception', 'Risk Accepted')
GROUP BY 
    CASE 
        WHEN CURRENT_DATE - discovered_date <= 30 THEN '0-30 days'
        WHEN CURRENT_DATE - discovered_date <= 60 THEN '31-60 days'
        WHEN CURRENT_DATE - discovered_date <= 90 THEN '61-90 days'
        WHEN CURRENT_DATE - discovered_date <= 180 THEN '91-180 days'
        ELSE '180+ days'
    END,
    severity
ORDER BY age_bucket, severity;

-- Team Performance Comparison
SELECT 
    a.owner_team,
    COUNT(*) FILTER (WHERE v.remediated_date IS NOT NULL) as remediated,
    COUNT(*) FILTER (WHERE v.remediated_date IS NULL) as open,
    ROUND(AVG(CASE WHEN v.remediated_date IS NOT NULL 
              THEN v.remediated_date - v.discovered_date END), 1) as avg_mttr,
    ROUND(100.0 * COUNT(*) FILTER (WHERE v.remediated_date <= v.sla_due_date) 
          / NULLIF(COUNT(*) FILTER (WHERE v.remediated_date IS NOT NULL), 0), 1) as sla_compliance_pct
FROM vulnerabilities v
JOIN assets a ON v.asset_id = a.asset_id
WHERE v.discovered_date >= CURRENT_DATE - 90
GROUP BY a.owner_team
ORDER BY open DESC;
