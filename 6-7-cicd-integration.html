Container Image Scanning

YAML 
# GitHub Actions: Container Scanning with Trivy
name: Container Security Scan

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  build-and-scan:
    runs-on: ubuntu-latest
    
    steps:
      - uses: actions/checkout@v4
      
      - name: Build Docker image
        run: docker build -t myapp:${{ github.sha }} .
      
      - name: Run Trivy vulnerability scanner
        uses: aquasecurity/trivy-action@master
        with:
          image-ref: 'myapp:${{ github.sha }}'
          format: 'sarif'
          output: 'trivy-results.sarif'
          severity: 'CRITICAL,HIGH'
          exit-code: '1'  # Fail build on findings
      
      - name: Upload scan results to GitHub Security
        uses: github/codeql-action/upload-sarif@v2
        with:
          sarif_file: 'trivy-results.sarif'
      
      - name: Push to registry (if scan passes)
        if: success()
        run: |
          docker tag myapp:${{ github.sha }} registry.company.com/myapp:${{ github.sha }}
          docker push registry.company.com/myapp:${{ github.sha }}

  stage: security-scan
  image: snyk/snyk:python
  script:
    - snyk test --severity-threshold=high
  allow_failure: false

container-scan:
  stage: security-scan
  image: aquasec/trivy:latest
  script:
    - trivy image --exit-code 1 --severity CRITICAL,HIGH $CONTAINER_IMAGE
  allow_failure: false

sast:
  stage: security-scan
  image: returntocorp/semgrep
  script:
    - semgrep --config=auto --error --severity=ERROR .
  allow_failure: false

deploy-staging:
  stage: deploy
  script:
    - kubectl apply -f k8s/staging/
  only:
    - develop
  needs:
    - dependency-scan
    - container-scan
    - sast

Policy as Code

Python 
"""
Security Gate Policy Evaluation
Determine build pass/fail based on scan results.
"""

from typing import Dict, List
from enum import Enum


class GateResult(Enum):
    PASS = "pass"
    WARN = "warn"
    FAIL = "fail"


def evaluate_security_gate(scan_results: List[Dict]) -> Dict:
    """
    Evaluate scan results against security policy.
    
    Args:
        scan_results: List of vulnerability findings
    
    Returns:
        Gate result with details
    """
    result = {
        'gate_result': GateResult.PASS,
        'critical_count': 0,
        'high_count': 0,
        'blocking_vulns': [],
        'warning_vulns': []
    }
    
    for vuln in scan_results:
        cvss = vuln.get('cvss_score', 0)
        severity = vuln.get('severity', 'Low')
        has_exploit = vuln.get('exploit_available', False)
        in_kev = vuln.get('cisa_kev', False)
        
        # FAIL conditions
        if in_kev:
            result['gate_result'] = GateResult.FAIL
            result['blocking_vulns'].append({
                'cve': vuln.get('cve_id'),
                'reason': 'CISA KEV - actively exploited'
            })
        elif cvss >= 9.0:
            result['gate_result'] = GateResult.FAIL
            result['critical_count'] += 1
            result['blocking_vulns'].append({
                'cve': vuln.get('cve_id'),
                'reason': f'Critical severity (CVSS {cvss})'
            })
        elif cvss >= 7.0 and has_exploit:
            result['gate_result'] = GateResult.FAIL
            result['blocking_vulns'].append({
                'cve': vuln.get('cve_id'),
                'reason': f'High severity with public exploit'
            })
        
        # WARN conditions
        elif cvss >= 7.0:
            if result['gate_result'] == GateResult.PASS:
                result['gate_result'] = GateResult.WARN
            result['high_count'] += 1
            result['warning_vulns'].append({
                'cve': vuln.get('cve_id'),
                'cvss': cvss
            })
    
    return result


# Pipeline integration
def security_gate_check(scan_file: str) -> int:
    """
    CLI entry point for CI/CD integration.
    Returns exit code: 0=pass, 1=fail, 2=warn
    """
    import json
    
    with open(scan_file) as f:
        results = json.load(f)
    
    evaluation = evaluate_security_gate(results)
    
    print(f"Gate Result: {evaluation['gate_result'].value.upper()}")
    print(f"Critical: {evaluation['critical_count']}, High: {evaluation['high_count']}")
    
    if evaluation['blocking_vulns']:
        print("\nBlocking vulnerabilities:")
        for v in evaluation['blocking_vulns']:
            print(f"  - {v['cve']}: {v['reason']}")
    
    exit_codes = {
        GateResult.PASS: 0,
        GateResult.WARN: 2,
        GateResult.FAIL: 1
    }
    
    return exit_codes[evaluation['gate_result']]

I define security gates that block deployment for critical vulnerabilities or CISA KEV findings while warning on high-severity issues. Summary

CI/CD integration shifts vulnerability detection left, finding issues before production deployment. Implement scanning at multiple stages with clear gate policies. Block on critical risks while allowing development velocity for lower-severity findings that can be tracked and remediated.
