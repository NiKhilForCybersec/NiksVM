"""
ServiceNow Change Request Automation
Auto-create change requests for vulnerability remediation.
"""

import requests
from typing import Dict, List
from datetime import datetime, timedelta


class ServiceNowChangeClient:
    """Create and manage change requests in ServiceNow."""
    
    def __init__(self, instance: str, username: str, password: str):
        self.base_url = f"https://{instance}.service-now.com/api/now"
        self.auth = (username, password)
        self.headers = {
            'Content-Type': 'application/json',
            'Accept': 'application/json'
        }
    
    def create_change_request(self, vuln: Dict, assets: List[Dict], 
                              priority: str) -> str:
        """
        Create change request for vulnerability remediation.
        
        Args:
            vuln: Vulnerability details
            assets: List of affected assets
            priority: P1-P5 priority level
        
        Returns:
            Change request number
        """
        # Determine change type based on priority
        change_type = self._determine_change_type(priority, assets)
        
        # Calculate dates
        dates = self._calculate_dates(priority)
        
        # Build description
        description = self._build_description(vuln, assets)
        
        payload = {
            'type': change_type,
            'short_description': f"Security Patch: {vuln.get('cve_id')} Remediation",
            'description': description,
            'category': 'Security',
            'priority': self._map_priority(priority),
            'risk': self._assess_risk(vuln, assets),
            'impact': self._assess_impact(assets),
            'start_date': dates['start'].isoformat(),
            'end_date': dates['end'].isoformat(),
            'assignment_group': 'Security Operations',
            'cmdb_ci': assets[0].get('cmdb_sys_id') if assets else None,
            'justification': f"Remediation of {vuln.get('cve_id')} - "
                           f"CVSS: {vuln.get('cvss_score', 'N/A')}",
            'implementation_plan': self._build_implementation_plan(vuln),
            'backout_plan': self._build_backout_plan(vuln),
            'test_plan': "Post-patch vulnerability rescan to confirm remediation"
        }
        
        response = requests.post(
            f"{self.base_url}/table/change_request",
            auth=self.auth,
            headers=self.headers,
            json=payload
        )
        response.raise_for_status()
        
        return response.json()['result']['number']
    
    def _determine_change_type(self, priority: str, assets: List[Dict]) -> str:
        """Determine change type based on priority and assets."""
        # Check for pre-approved standard change eligibility
        if priority in ['P4', 'P5']:
            all_low_tier = all(
                a.get('criticality_tier', 3) >= 3 for a in assets
            )
            if all_low_tier:
                return 'standard'
        
        if priority == 'P1':
            return 'emergency'
        
        return 'normal'
    
    def _calculate_dates(self, priority: str) -> Dict:
        """Calculate change window dates."""
        now = datetime.now()
        
        windows = {
            'P1': (timedelta(hours=4), timedelta(hours=8)),
            'P2': (timedelta(days=3), timedelta(days=4)),
            'P3': (timedelta(days=7), timedelta(days=8)),
            'P4': (timedelta(days=14), timedelta(days=15)),
            'P5': (timedelta(days=30), timedelta(days=31))
        }
        
        start_delta, end_delta = windows.get(priority, (timedelta(days=7), timedelta(days=8)))
        
        return {
            'start': now + start_delta,
            'end': now + end_delta
        }
    
    def _map_priority(self, priority: str) -> str:
        """Map VM priority to ServiceNow priority."""
        mapping = {
            'P1': '1',  # Critical
            'P2': '2',  # High
            'P3': '3',  # Moderate
            'P4': '4',  # Low
            'P5': '5'   # Planning
        }
        return mapping.get(priority, '3')
    
    def _assess_risk(self, vuln: Dict, assets: List[Dict]) -> str:
        """Assess change risk level."""
        cvss = vuln.get('cvss_score', 5.0)
        tier_1_count = sum(1 for a in assets if a.get('criticality_tier') == 1)
        
        if tier_1_count > 0:
            return 'high'
        elif cvss >= 9.0 or len(assets) > 50:
            return 'high'
        elif cvss >= 7.0:
            return 'moderate'
        else:
            return 'low'
    
    def _assess_impact(self, assets: List[Dict]) -> str:
        """Assess change impact."""
        tier_1_count = sum(1 for a in assets if a.get('criticality_tier') == 1)
        
        if tier_1_count > 0 or len(assets) > 100:
            return 'high'
        elif len(assets) > 20:
            return 'medium'
        else:
            return 'low'
    
    def _build_description(self, vuln: Dict, assets: List[Dict]) -> str:
        """Build change description."""
        asset_list = '\n'.join(f"- {a.get('hostname')}" for a in assets[:10])
        if len(assets) > 10:
            asset_list += f"\n- ... and {len(assets) - 10} more"
        
        return f"""
Security Vulnerability Remediation

CVE: {vuln.get('cve_id')}
CVSS Score: {vuln.get('cvss_score', 'N/A')}
Severity: {vuln.get('severity', 'N/A')}

Description:
{vuln.get('description', 'See vendor advisory')}

Affected Assets ({len(assets)} total):
{asset_list}

Remediation:
{vuln.get('solution', 'Apply vendor-provided patch')}
        """
    
    def _build_implementation_plan(self, vuln: Dict) -> str:
        """Build implementation plan."""
        return f"""
1. Create backup/snapshot of affected systems
2. Verify patch availability: {vuln.get('patch_id', 'See advisory')}
3. Deploy patch to pilot group (if applicable)
4. Deploy patch to production systems
5. Reboot systems (if required)
6. Verify system functionality
7. Run vulnerability rescan
8. Close remediation ticket
        """
    
    def _build_backout_plan(self, vuln: Dict) -> str:
        """Build backout/rollback plan."""
        return """
1. Restore from pre-patch backup/snapshot
2. Verify system functionality restored
3. Document rollback reason
4. Notify security team of failed remediation
5. Implement compensating controls if needed
6. Schedule remediation retry
        """


# Example usage

print(f"Created change request: {change_number}")

Summary
