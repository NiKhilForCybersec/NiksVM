"""
Business Impact Quantification
Translate business context into priority scores.
"""

from typing import Dict
from dataclasses import dataclass
from enum import Enum


class ImpactLevel(Enum):
    CRITICAL = 4  # >$10M or catastrophic
    HIGH = 3      # $1M-$10M
    MEDIUM = 2    # $100K-$1M
    LOW = 1       # <$100K


@dataclass
class BusinessImpact:
    financial_impact: ImpactLevel
    operational_impact: ImpactLevel
    reputational_impact: ImpactLevel
    compliance_impact: ImpactLevel
    data_impact: ImpactLevel
    
    def calculate_score(self) -> float:
        """Calculate weighted business impact score (0-100)."""
        weights = {
            'financial_impact': 0.25,
            'operational_impact': 0.25,
            'reputational_impact': 0.15,
            'compliance_impact': 0.20,
            'data_impact': 0.15
        }
        
        score = 0
        for field, weight in weights.items():
            impact = getattr(self, field)
            # Normalize to 0-100: (value / max) * 100
            normalized = (impact.value / 4) * 100
            score += normalized * weight
        
        return round(score, 1)


class BusinessContextAnalyzer:
    """Analyze business context for vulnerability prioritization."""
    
    # Application criticality mappings
    APP_CRITICALITY = {
        'payment_processing': ImpactLevel.CRITICAL,
        'customer_portal': ImpactLevel.CRITICAL,
        'erp': ImpactLevel.HIGH,
        'crm': ImpactLevel.HIGH,
        'email': ImpactLevel.HIGH,
        'file_server': ImpactLevel.MEDIUM,
        'internal_app': ImpactLevel.MEDIUM,
        'development': ImpactLevel.LOW,
        'test': ImpactLevel.LOW
    }
    
    # Data classification impact
    DATA_IMPACT = {
        'restricted': ImpactLevel.CRITICAL,    # PII, PHI, financial
        'confidential': ImpactLevel.HIGH,      # Internal sensitive
        'internal': ImpactLevel.MEDIUM,        # Internal use
        'public': ImpactLevel.LOW              # Public data
    }
    
    # Compliance scope impact
    COMPLIANCE_IMPACT = {
        'pci_dss': ImpactLevel.CRITICAL,
        'hipaa': ImpactLevel.CRITICAL,
        'sox': ImpactLevel.HIGH,
        'gdpr': ImpactLevel.HIGH,
        'none': ImpactLevel.LOW
    }
    
    def analyze_asset(self, asset: Dict) -> BusinessImpact:
        """
        Analyze business impact for an asset.
        
        Args:
            asset: Asset with business context attributes
        
        Returns:
            BusinessImpact assessment
        """
        # Determine financial impact based on application type
        app_type = asset.get('application_type', 'internal_app')
        financial = self.APP_CRITICALITY.get(app_type, ImpactLevel.MEDIUM)
        
        # Revenue-generating assets get elevated financial impact
        if asset.get('revenue_generating', False):
            financial = ImpactLevel.CRITICAL
        
        # Operational impact based on user count and dependency
        user_count = asset.get('user_count', 0)
        if user_count > 10000:
            operational = ImpactLevel.CRITICAL
        elif user_count > 1000:
            operational = ImpactLevel.HIGH
        elif user_count > 100:
            operational = ImpactLevel.MEDIUM
        else:
            operational = ImpactLevel.LOW
        
        # Reputational impact for customer-facing assets
        if asset.get('customer_facing', False):
            reputational = ImpactLevel.HIGH
            if asset.get('internet_facing', False):
                reputational = ImpactLevel.CRITICAL
        else:
            reputational = ImpactLevel.LOW
        
        # Compliance impact
        compliance_scopes = asset.get('compliance_scope', [])
        compliance = ImpactLevel.LOW
        for scope in compliance_scopes:
            scope_impact = self.COMPLIANCE_IMPACT.get(scope.lower(), ImpactLevel.LOW)
            if scope_impact.value > compliance.value:
                compliance = scope_impact
        
        # Data impact
        data_class = asset.get('data_classification', 'internal')
        data = self.DATA_IMPACT.get(data_class.lower(), ImpactLevel.MEDIUM)
        
        return BusinessImpact(
            financial_impact=financial,
            operational_impact=operational,
            reputational_impact=reputational,
            compliance_impact=compliance,
            data_impact=data
        )


# Example usage
analyzer = BusinessContextAnalyzer()

asset = {
    'name': 'payment-gateway-01',
    'application_type': 'payment_processing',
    'revenue_generating': True,
    'customer_facing': True,
    'internet_facing': True,
    'user_count': 50000,
    'compliance_scope': ['pci_dss', 'sox'],
    'data_classification': 'restricted'
}

impact = analyzer.analyze_asset(asset)
score = impact.calculate_score()

print(f"Business Impact Score: {score}/100")
print(f"Financial: {impact.financial_impact.name}")
print(f"Operational: {impact.operational_impact.name}")
print(f"Compliance: {impact.compliance_impact.name}")
