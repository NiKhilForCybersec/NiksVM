This page covers vulnerability prioritization strategies including why CVSS alone is insufficient, modern risk-based prioritization frameworks, exploitation probability, and building effective prioritization models for your organization.

The Prioritization Problem

Organizations face thousands of vulnerabilities but limited resources to fix them. Effective prioritization ensures you fix the vulnerabilities most likely to be exploited against your most critical assets first.

"""
Vulnerability Prioritization Engine
Risk-based prioritization considering multiple factors.
"""

from typing import Dict, List, Optional
from dataclasses import dataclass
from datetime import datetime, timedelta
from enum import Enum


class PriorityLevel(Enum):
    P1_CRITICAL = 1
    P2_HIGH = 2
    P3_MEDIUM = 3
    P4_LOW = 4


@dataclass
class PrioritizedVulnerability:
    vuln_id: str
    cve_id: str
    asset_id: str
    priority: PriorityLevel
    risk_score: float
    factors: Dict[str, float]
    sla_days: int
    due_date: datetime


class VulnerabilityPrioritizer:
    """Calculate risk-based priority for vulnerabilities."""
    
    # Severity score mapping
    SEVERITY_SCORES = {
        'Critical': 1.0,
        'High': 0.7,
        'Medium': 0.4,
        'Low': 0.1,
        'Info': 0.0
    }
    
    # Asset tier multipliers
    TIER_MULTIPLIERS = {
        1: 1.0,   # Critical assets
        2: 0.7,   # High importance
        3: 0.4,   # Medium importance
        4: 0.2    # Low importance
    }
    
    # Network exposure multipliers
    EXPOSURE_MULTIPLIERS = {
        'internet_facing': 1.5,
        'dmz': 1.3,
        'internal': 1.0,
        'isolated': 0.8
    }
    
    # SLA by priority (days)
    SLA_DAYS = {
        PriorityLevel.P1_CRITICAL: 7,
        PriorityLevel.P2_HIGH: 14,
        PriorityLevel.P3_MEDIUM: 30,
        PriorityLevel.P4_LOW: 90
    }
    
    def __init__(self, kev_list: List[str] = None, 
                 exploit_db: Dict[str, bool] = None):
        """
        Initialize prioritizer with threat intelligence.
        
        Args:
            kev_list: List of CVEs in CISA KEV
            exploit_db: Dict of CVE -> has_public_exploit
        """
        self.kev_list = set(kev_list or [])
        self.exploit_db = exploit_db or {}
    
    def calculate_severity_score(self, vuln: Dict) -> float:
        """Calculate severity component (20% weight)."""
        severity = vuln.get('severity', 'Medium')
        return self.SEVERITY_SCORES.get(severity, 0.4) * 0.20
    
    def calculate_exploitability_score(self, vuln: Dict) -> float:
        """Calculate exploitability component (35% weight)."""
        score = 0.0
        cve_id = vuln.get('cve_id')
        
        # CISA KEV (highest indicator)
        if cve_id in self.kev_list:
            score += 0.5
        
        # Public exploit available
        if self.exploit_db.get(cve_id, False):
            score += 0.3
        
        # Exploit maturity from scanner
        maturity = vuln.get('exploit_maturity', '').lower()
        if 'functional' in maturity or 'high' in maturity:
            score += 0.2
        elif 'poc' in maturity:
            score += 0.1
        
        # Active exploitation flag
        if vuln.get('actively_exploited'):
            score += 0.4
        
        # Age decay (older unexploited vulns less urgent)
        published_date = vuln.get('published_date')
        if published_date:
            age_days = (datetime.now() - published_date).days
            age_penalty = min(age_days / 365, 0.3)  # Max 0.3 reduction
            score -= age_penalty
        
        return max(score, 0) * 0.35
    
    def calculate_asset_score(self, asset: Dict) -> float:
        """Calculate asset criticality component (30% weight)."""
        tier = asset.get('criticality_tier', 3)
        exposure = asset.get('network_zone', 'internal')
        
        base_score = self.TIER_MULTIPLIERS.get(tier, 0.4)
        exposure_mult = self.EXPOSURE_MULTIPLIERS.get(exposure, 1.0)
        
        return base_score * exposure_mult * 0.30
    
    def calculate_threat_score(self, vuln: Dict) -> float:
        """Calculate threat context component (15% weight)."""
        score = 0.0
        
        if vuln.get('apt_targeted'):
            score += 0.3
        if vuln.get('ransomware_used'):
            score += 0.4
        if vuln.get('industry_targeted'):
            score += 0.2
        if vuln.get('exploitation_surge'):
            score += 0.2
        
        return min(score, 1.0) * 0.15
    
    def prioritize(self, vuln: Dict, asset: Dict) -> PrioritizedVulnerability:
        """
        Calculate priority for a vulnerability on a specific asset.
        
        Args:
            vuln: Vulnerability data
            asset: Asset data
            
        Returns:
            PrioritizedVulnerability with risk score and priority level
        """
        # Calculate component scores
        severity = self.calculate_severity_score(vuln)
        exploitability = self.calculate_exploitability_score(vuln)
        asset_crit = self.calculate_asset_score(asset)
        threat = self.calculate_threat_score(vuln)
        
        # Total risk score (0-1)
        risk_score = severity + exploitability + asset_crit + threat
        
        # Determine priority level
        if risk_score >= 0.7:
            priority = PriorityLevel.P1_CRITICAL
        elif risk_score >= 0.5:
            priority = PriorityLevel.P2_HIGH
        elif risk_score >= 0.3:
            priority = PriorityLevel.P3_MEDIUM
        else:
            priority = PriorityLevel.P4_LOW
        
        # Override: KEV vulns on Tier 1/2 assets always P1
        if vuln.get('cve_id') in self.kev_list and asset.get('criticality_tier', 3) <= 2:
            priority = PriorityLevel.P1_CRITICAL
            risk_score = max(risk_score, 0.8)
        
        sla = self.SLA_DAYS[priority]
        due_date = datetime.now() + timedelta(days=sla)
        
        return PrioritizedVulnerability(
            vuln_id=vuln.get('id', 'unknown'),
            cve_id=vuln.get('cve_id'),
            asset_id=asset.get('id'),
            priority=priority,
            risk_score=round(risk_score, 3),
            factors={
                'severity': round(severity, 3),
                'exploitability': round(exploitability, 3),
                'asset_criticality': round(asset_crit, 3),
                'threat_context': round(threat, 3)
            },
            sla_days=sla,
            due_date=due_date
        )
    
    def bulk_prioritize(self, vulnerabilities: List[Dict], 
                        assets: Dict[str, Dict]) -> List[PrioritizedVulnerability]:
        """Prioritize multiple vulnerabilities."""
        results = []
        
        for vuln in vulnerabilities:
            asset_id = vuln.get('asset_id')
            asset = assets.get(asset_id, {})
            
            result = self.prioritize(vuln, asset)
            results.append(result)
        
        # Sort by risk score descending
        results.sort(key=lambda x: x.risk_score, reverse=True)
        
        return results


# Example usage
kev_cves = ['CVE-2021-44228', 'CVE-2023-23397', 'CVE-2024-21762']
exploit_db = {'CVE-2021-44228': True, 'CVE-2023-23397': True}

prioritizer = VulnerabilityPrioritizer(kev_list=kev_cves, exploit_db=exploit_db)

vuln = {
    'id': 'vuln-001',
    'cve_id': 'CVE-2021-44228',
    'severity': 'Critical',
    'actively_exploited': True,
    'published_date': datetime(2021, 12, 10)
}

asset = {
    'id': 'srv-web-01',
    'criticality_tier': 1,
    'network_zone': 'internet_facing'
}

result = prioritizer.prioritize(vuln, asset)
print(f"Priority: {result.priority.name}")
print(f"Risk Score: {result.risk_score}")
print(f"Due Date: {result.due_date.strftime('%Y-%m-%d')}")

"How do you prioritize vulnerabilities when you have thousands to address?"

Sample answer: "I use a risk-based approach that goes beyond CVSS scores. My prioritization model considers four factors: CVSS severity (20%), exploitability including CISA KEV status and public exploits (35%), asset criticality from our tiering model (30%), and threat context like ransomware usage (15%). This ensures a critical vulnerability on a dev test server doesn't get prioritized over a high vulnerability on an internet-facing production database. I also maintain an automated system that recalculates priority as threat intelligence changes."

Summary

Effective vulnerability prioritization combines vulnerability severity with exploitation likelihood, asset criticality, and threat context. Move beyond CVSS-only prioritization to ensure you're fixing the vulnerabilities that matter most to your organization first.
