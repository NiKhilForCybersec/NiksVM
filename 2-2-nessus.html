Installation and Setup

Linux Installation

Bash 
# Download Nessus for your distribution
# Ubuntu/Debian
curl -o Nessus-10.x.x-ubuntu1404_amd64.deb \
  "https://www.tenable.com/downloads/api/v1/public/pages/nessus/downloads/xxxxx/download?i_agree_to_tenable_license_agreement=true"

# Install the package
sudo dpkg -i Nessus-10.x.x-ubuntu1404_amd64.deb

# Start Nessus service
sudo systemctl start nessusd
sudo systemctl enable nessusd

# Check service status
sudo systemctl status nessusd

# Access web interface (default port 8834)
# https://localhost:8834

# RHEL/CentOS
sudo yum install Nessus-10.x.x-es7.x86_64.rpm
sudo systemctl start nessusd

Enabling/Disabling Plugin Families

Python 
# Example: Configure plugin families via API
import requests
import json

class NessusAPI:
    def __init__(self, url, access_key, secret_key):
        self.url = url
        self.headers = {
            'X-ApiKeys': f'accessKey={access_key}; secretKey={secret_key}',
            'Content-Type': 'application/json'
        }
    
    def get_plugin_families(self):
        """Get all plugin families."""
        response = requests.get(
            f'{self.url}/plugins/families',
            headers=self.headers,
            verify=False  # Set True in production with valid cert
        )
        return response.json()
    
    def get_family_plugins(self, family_id):
        """Get plugins in a specific family."""
        response = requests.get(
            f'{self.url}/plugins/families/{family_id}',
            headers=self.headers,
            verify=False
        )
        return response.json()

# Example: List Windows plugin families
nessus = NessusAPI(
    'https://nessus.company.com:8834',
    'your-access-key',
    'your-secret-key'
)

families = nessus.get_plugin_families()
for family in families['families']:
    if 'Windows' in family['name']:
        print(f"ID: {family['id']}, Name: {family['name']}, Count: {family['count']}")

API Integration

Nessus provides a REST API for automation. Here's a comprehensive Python wrapper for common operations:

Python 
"""
Nessus API Integration Module
Provides comprehensive scanning automation capabilities.
"""

import requests
import time
import json
from typing import Dict, List, Optional
import urllib3

# Disable SSL warnings for self-signed certs (not for production)
urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)


class NessusScanner:
    """
    Nessus API client for vulnerability scanning automation.
    
    Usage:
        scanner = NessusScanner('https://nessus:8834', 'access_key', 'secret_key')
        scan_id = scanner.create_scan('Weekly Scan', '192.168.1.0/24', policy_id=1)
        scanner.launch_scan(scan_id)
        scanner.wait_for_scan(scan_id)
        results = scanner.get_scan_results(scan_id)
    """
    
    def __init__(self, url: str, access_key: str, secret_key: str, verify_ssl: bool = False):
        self.url = url.rstrip('/')
        self.verify_ssl = verify_ssl
        self.headers = {
            'X-ApiKeys': f'accessKey={access_key}; secretKey={secret_key}',
            'Content-Type': 'application/json'
        }
    
    def _request(self, method: str, endpoint: str, data: Dict = None) -> Dict:
        """Make API request with error handling."""
        url = f'{self.url}{endpoint}'
        
        try:
            if method == 'GET':
                response = requests.get(url, headers=self.headers, verify=self.verify_ssl)
            elif method == 'POST':
                response = requests.post(url, headers=self.headers, 
                                        json=data, verify=self.verify_ssl)
            elif method == 'PUT':
                response = requests.put(url, headers=self.headers,
                                       json=data, verify=self.verify_ssl)
            elif method == 'DELETE':
                response = requests.delete(url, headers=self.headers, verify=self.verify_ssl)
            
            response.raise_for_status()
            return response.json() if response.content else {}
            
        except requests.exceptions.RequestException as e:
            raise Exception(f"API request failed: {e}")
    
    # â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
    # SCAN MANAGEMENT
    # â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
    
    def list_scans(self) -> List[Dict]:
        """List all scans."""
        result = self._request('GET', '/scans')
        return result.get('scans', [])
    
    def create_scan(self, name: str, targets: str, policy_id: int = None,
                    folder_id: int = None, description: str = '') -> int:
        """
        Create a new scan.
        
        Args:
            name: Scan name
            targets: Comma-separated IPs, ranges, or hostnames
            policy_id: Policy template ID
            folder_id: Folder to store scan
            description: Scan description
            
        Returns:
            Scan ID
        """
        settings = {
            'name': name,
            'text_targets': targets,
            'description': description
        }
        
        if policy_id:
            settings['policy_id'] = policy_id
        if folder_id:
            settings['folder_id'] = folder_id
        
        # Use basic network scan template if no policy specified
        data = {
            'uuid': 'ab4bacd2-05f6-425c-9d79-3b5f8e84f2b0',  # Basic Network Scan
            'settings': settings
        }
        
        result = self._request('POST', '/scans', data)
        return result['scan']['id']
    
    def launch_scan(self, scan_id: int) -> str:
        """Launch a scan and return scan UUID."""
        result = self._request('POST', f'/scans/{scan_id}/launch')
        return result.get('scan_uuid')
    
    def pause_scan(self, scan_id: int) -> None:
        """Pause a running scan."""
        self._request('POST', f'/scans/{scan_id}/pause')
    
    def resume_scan(self, scan_id: int) -> None:
        """Resume a paused scan."""
        self._request('POST', f'/scans/{scan_id}/resume')
    
    def stop_scan(self, scan_id: int) -> None:
        """Stop a running scan."""
        self._request('POST', f'/scans/{scan_id}/stop')
    
    def delete_scan(self, scan_id: int) -> None:
        """Delete a scan."""
        self._request('DELETE', f'/scans/{scan_id}')
    
    def get_scan_status(self, scan_id: int) -> str:
        """Get current scan status."""
        result = self._request('GET', f'/scans/{scan_id}')
        return result['info']['status']
    
    def wait_for_scan(self, scan_id: int, poll_interval: int = 30, 
                      timeout: int = 3600) -> bool:
        """
        Wait for scan to complete.
        
        Args:
            scan_id: Scan ID to monitor
            poll_interval: Seconds between status checks
            timeout: Maximum wait time in seconds
            
        Returns:
            True if scan completed, False if timed out
        """
        start_time = time.time()
        
        while time.time() - start_time < timeout:
            status = self.get_scan_status(scan_id)
            
            if status == 'completed':
                return True
            elif status in ['canceled', 'aborted']:
                raise Exception(f"Scan {scan_id} was {status}")
            
            print(f"Scan status: {status}, waiting...")
            time.sleep(poll_interval)
        
        return False
    
    # â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
    # RESULTS AND REPORTING
    # â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
    
    def get_scan_results(self, scan_id: int) -> Dict:
        """Get detailed scan results."""
        return self._request('GET', f'/scans/{scan_id}')
    
    def get_host_vulnerabilities(self, scan_id: int, host_id: int) -> List[Dict]:
        """Get vulnerabilities for a specific host."""
        result = self._request('GET', f'/scans/{scan_id}/hosts/{host_id}')
        return result.get('vulnerabilities', [])
    
    def get_vulnerability_details(self, scan_id: int, host_id: int, 
                                   plugin_id: int) -> Dict:
        """Get detailed info about a specific vulnerability."""
        return self._request('GET', 
            f'/scans/{scan_id}/hosts/{host_id}/plugins/{plugin_id}')
    
    def export_scan(self, scan_id: int, format: str = 'nessus') -> bytes:
        """
        Export scan results.
        
        Args:
            scan_id: Scan ID
            format: 'nessus', 'csv', 'pdf', 'html', 'db'
            
        Returns:
            Exported file content as bytes
        """
        # Initiate export
        data = {'format': format}
        result = self._request('POST', f'/scans/{scan_id}/export', data)
        file_id = result['file']
        
        # Wait for export to complete
        while True:
            status = self._request('GET', f'/scans/{scan_id}/export/{file_id}/status')
            if status['status'] == 'ready':
                break
            time.sleep(2)
        
        # Download export
        download_url = f'{self.url}/scans/{scan_id}/export/{file_id}/download'
        response = requests.get(download_url, headers=self.headers, 
                               verify=self.verify_ssl)
        return response.content
    
    def get_vulnerability_summary(self, scan_id: int) -> Dict:
        """Get vulnerability count summary by severity."""
        results = self.get_scan_results(scan_id)
        
        summary = {
            'critical': 0,
            'high': 0,
            'medium': 0,
            'low': 0,
            'info': 0,
            'hosts_scanned': len(results.get('hosts', []))
        }
        
        for host in results.get('hosts', []):
            summary['critical'] += host.get('critical', 0)
            summary['high'] += host.get('high', 0)
            summary['medium'] += host.get('medium', 0)
            summary['low'] += host.get('low', 0)
            summary['info'] += host.get('info', 0)
        
        return summary
