This page covers authenticated vulnerability scanning including credential types, service account configuration for Windows and Linux, network device authentication, database scanning, and secure credential management practices.

Why Authenticated Scanning Matters

Authenticated scans detect 60-70% more vulnerabilities than unauthenticated scans. Without credentials, scanners only see externally visible information like banners and open ports.

Windows Authentication

PowerShell 
# Create Vulnerability Scanner Service Account in Active Directory

# Create OU for service accounts
New-ADOrganizationalUnit -Name "Service Accounts" -Path "DC=corp,DC=company,DC=com"

# Create scanner service account
$Password = ConvertTo-SecureString "ComplexP@ssw0rd!" -AsPlainText -Force
New-ADUser -Name "svc_vuln_scan" `
    -SamAccountName "svc_vuln_scan" `
    -UserPrincipalName "svc_vuln_scan@corp.company.com" `
    -Path "OU=Service Accounts,DC=corp,DC=company,DC=com" `
    -AccountPassword $Password `
    -Enabled $true `
    -PasswordNeverExpires $true `
    -CannotChangePassword $true `
    -Description "Vulnerability Scanner Service Account"

# Add to local Administrators on target systems via GPO
# Or create a dedicated group
New-ADGroup -Name "Vulnerability Scanners" `
    -GroupScope Global `
    -GroupCategory Security `
    -Path "OU=Security Groups,DC=corp,DC=company,DC=com"

Add-ADGroupMember -Identity "Vulnerability Scanners" -Members "svc_vuln_scan"

Windows Target Configuration

PowerShell 
# Configure Windows targets for authenticated scanning

# 1. Enable Remote Registry service (for WMI scanning)
Set-Service -Name "RemoteRegistry" -StartupType Automatic
Start-Service -Name "RemoteRegistry"

# 2. Configure Windows Firewall (if enabled)
# Enable File and Printer Sharing
Enable-NetFirewallRule -DisplayGroup "File and Printer Sharing"

# Enable WMI
Enable-NetFirewallRule -DisplayGroup "Windows Management Instrumentation (WMI)"

# 3. Enable WinRM (for PowerShell-based scanning)
Enable-PSRemoting -Force
Set-Item WSMan:\localhost\Client\TrustedHosts -Value "scanner.company.com"

# 4. Configure UAC for remote admin access
# Required for non-domain admin local accounts
$RegPath = "HKLM:\SOFTWARE\Microsoft\Windows\CurrentVersion\Policies\System"
Set-ItemProperty -Path $RegPath -Name "LocalAccountTokenFilterPolicy" -Value 1

# 5. Verify admin share access
# Test from scanner: net use \\target\c$ /user:domain\svc_vuln_scan

# GPO Settings for domain-wide deployment:
# Computer Configuration > Preferences > Windows Settings > Registry
# Add LocalAccountTokenFilterPolicy = 1
# Computer Configuration > Policies > Admin Templates > Network > Network Connections > Windows Firewall
# Allow inbound file and printer sharing exception

Linux/Unix Authentication

Bash 
#!/bin/bash
# Configure Linux targets for authenticated scanning

# 1. Create scanner service account
sudo useradd -m -s /bin/bash -c "Vulnerability Scanner" nessus_scan
sudo passwd nessus_scan  # Set strong password

# 2. Configure sudo access (choose one approach)

# Option A: Full sudo (not recommended for production)
echo "nessus_scan ALL=(ALL) NOPASSWD: ALL" | sudo tee /etc/sudoers.d/nessus_scan

# Option B: Limited sudo (recommended)
cat << 'EOF' | sudo tee /etc/sudoers.d/nessus_scan
# Vulnerability Scanner - Limited sudo access
nessus_scan ALL=(root) NOPASSWD: /bin/cat /etc/shadow
nessus_scan ALL=(root) NOPASSWD: /bin/cat /etc/passwd
nessus_scan ALL=(root) NOPASSWD: /usr/bin/find
nessus_scan ALL=(root) NOPASSWD: /bin/ls
nessus_scan ALL=(root) NOPASSWD: /usr/bin/rpm -qa
nessus_scan ALL=(root) NOPASSWD: /usr/bin/dpkg -l
nessus_scan ALL=(root) NOPASSWD: /usr/sbin/dmidecode
nessus_scan ALL=(root) NOPASSWD: /sbin/iptables -L
nessus_scan ALL=(root) NOPASSWD: /bin/netstat
nessus_scan ALL=(root) NOPASSWD: /bin/ss
nessus_scan ALL=(root) NOPASSWD: /usr/bin/systemctl list-units
EOF

sudo chmod 440 /etc/sudoers.d/nessus_scan

# 3. SSH Key Authentication (recommended over password)
# On scanner server:
ssh-keygen -t ed25519 -f /opt/scanner/keys/nessus_scan -N ""

# Copy public key to targets:
ssh-copy-id -i /opt/scanner/keys/nessus_scan.pub nessus_scan@target

# 4. Verify SSH configuration allows key auth
grep -E "^PubkeyAuthentication|^PasswordAuthentication" /etc/ssh/sshd_config
# PubkeyAuthentication yes

# 5. Test connectivity
ssh -i /opt/scanner/keys/nessus_scan nessus_scan@target "sudo cat /etc/shadow | head -1"
