import requests
from typing import Dict, List


class SlackNotifier:
    """Send vulnerability notifications to Slack."""
    
    def __init__(self, webhook_url: str, bot_token: str = None):
        self.webhook_url = webhook_url
        self.bot_token = bot_token
    
    def send_critical_alert(self, vuln: Dict, assets: List[Dict]):
        """Send critical vulnerability alert."""
        
        asset_list = '\n'.join(f"â€¢ {a['hostname']}" for a in assets[:10])
        
        payload = {
            'blocks': [
                {
                    'type': 'header',
                    'text': {
                        'type': 'plain_text',
                        'text': 'ğŸš¨ Critical Vulnerability Alert',
                        'emoji': True
                    }
                },
                {
                    'type': 'section',
                    'fields': [
                        {'type': 'mrkdwn', 'text': f"*CVE:*\n{vuln.get('cve_id')}"},
                        {'type': 'mrkdwn', 'text': f"*CVSS:*\n{vuln.get('cvss_score')}"},
                        {'type': 'mrkdwn', 'text': f"*CISA KEV:*\n{'Yes âš ï¸' if vuln.get('cisa_kev') else 'No'}"},
                        {'type': 'mrkdwn', 'text': f"*Assets:*\n{len(assets)}"}
                    ]
                },
                {
                    'type': 'section',
                    'text': {
                        'type': 'mrkdwn',
                        'text': f"*Affected Systems:*\n{asset_list}"
                    }

                {

import requests
from typing import Dict, List


class TeamsNotifier:
    """Send vulnerability notifications to Microsoft Teams."""
    
    def __init__(self, webhook_url: str):

    def send_critical_alert(self, vuln: Dict, assets: List[Dict]):

        payload = {
            '@type': 'MessageCard',
            '@context': 'http://schema.org/extensions',
            'themeColor': 'FF0000',
            'summary': f"Critical Vulnerability: {vuln.get('cve_id')}",
            'sections': [
                {
                    'activityTitle': f"ğŸš¨ Critical Vulnerability Alert",
                    'facts': [
                        {'name': 'CVE', 'value': vuln.get('cve_id')},
                        {'name': 'CVSS Score', 'value': str(vuln.get('cvss_score'))},
                        {'name': 'CISA KEV', 'value': 'Yes âš ï¸' if vuln.get('cisa_kev') else 'No'},
                        {'name': 'Affected Assets', 'value': str(len(assets))},
                        {'name': 'Priority', 'value': vuln.get('priority', 'P1')}
                    ],
                    'markdown': True
                }
            ],
            'potentialAction': [
                {
                    '@type': 'OpenUri',
                    'name': 'View Details',
                    'targets': [{'os': 'default', 'uri': vuln.get('scanner_url', '#')}]
                }
            ]
        }
        
        response = requests.post(self.webhook_url, json=payload)
        return response.status_code == 200

ğŸ’¡ Interview Question

"How do you handle vulnerability notifications without causing alert fatigue?"

Sample answer: "I tier notifications to match urgency. Immediate alerts go only for P1 vulnerabilities and CISA KEV additions to the security team. SLA warnings go directly to asset owners as due dates approach. Daily digests summarize new findings without flooding channels. I also aggregate by CVE rather than per-instance, so one vulnerability affecting 100 servers generates one notification with a count, not 100 separate alerts. This keeps notifications actionable rather than noise."

Summary

Effective notifications balance urgency with avoiding alert fatigue. Tier by severity, aggregate intelligently, and route to appropriate recipients. Make every notification actionable with clear context and direct links.
