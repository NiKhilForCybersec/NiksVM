"""
Qualys API Integration Examples
"""

import requests
from typing import Dict, List
import xml.etree.ElementTree as ET


class QualysAPI:
    """Qualys API client."""
    
    def __init__(self, username: str, password: str, 
                 platform: str = 'qualysapi.qualys.com'):
        self.base_url = f"https://{platform}"
        self.auth = (username, password)
        self.headers = {'X-Requested-With': 'Python'}
    
    def get_host_detections(self, severities: List[int] = None) -> List[Dict]:
        """Get vulnerability detections from hosts."""
        
        params = {
            'action': 'list',
            'show_results': '1',
            'status': 'New,Active,Re-Opened'
        }
        
        if severities:
            params['severities'] = ','.join(map(str, severities))
        
        response = requests.post(
            f"{self.base_url}/api/2.0/fo/asset/host/vm/detection/",
            auth=self.auth,
            headers=self.headers,
            data=params
        )
        
        # Parse XML response
        root = ET.fromstring(response.text)
        detections = []
        
        for host in root.findall('.//HOST'):
            for detection in host.findall('.//DETECTION'):
                detections.append({
                    'host_ip': host.findtext('IP'),
                    'hostname': host.findtext('DNS'),
                    'qid': detection.findtext('QID'),
                    'severity': detection.findtext('SEVERITY'),
                    'status': detection.findtext('STATUS'),
                    'first_found': detection.findtext('FIRST_FOUND_DATETIME'),
                    'last_found': detection.findtext('LAST_FOUND_DATETIME')
                })
        
        return detections
    
    def get_qid_details(self, qids: List[int]) -> List[Dict]:
        """Get knowledge base details for QIDs."""
        
        params = {
            'action': 'list',
            'ids': ','.join(map(str, qids)),
            'details': 'All'
        }
        
        response = requests.post(
            f"{self.base_url}/api/2.0/fo/knowledge_base/vuln/",
            auth=self.auth,
            headers=self.headers,
            data=params
        )
        
        root = ET.fromstring(response.text)
        vulns = []
        
        for vuln in root.findall('.//VULN'):
            vulns.append({
                'qid': vuln.findtext('QID'),
                'title': vuln.findtext('TITLE'),
                'severity': vuln.findtext('SEVERITY_LEVEL'),
                'cvss_base': vuln.findtext('.//CVSS/BASE'),
                'cves': [cve.text for cve in vuln.findall('.//CVE/ID')],
                'solution': vuln.findtext('SOLUTION'),
                'patchable': vuln.findtext('PATCHABLE') == '1'
            })
        
        return vulns
    
    def launch_scan(self, scan_ref: str) -> str:
        """Launch a vulnerability scan."""
        
        params = {
            'action': 'launch',
            'scan_ref': scan_ref
        }
        
        response = requests.post(
            f"{self.base_url}/api/2.0/fo/scan/",
            auth=self.auth,
            headers=self.headers,
            data=params
        )
        
        return response.text

"What's the difference between Qualys QID and CVE?"

Sample answer: "QID is Qualys's internal identifier for vulnerabilities. One QID might map to multiple CVEs if a single check detects several related vulnerabilities. Conversely, one CVE might have multiple QIDs for different vendors or platforms. QIDs also include findings that aren't CVEsâ€”like misconfigurations or informational checks. When reporting or integrating with other tools, I use the CVE mapping for standardization, but QID is what Qualys uses for tracking and remediation."

Qualys VMDR provides comprehensive vulnerability management with TruRisk for intelligent prioritization. The QID system offers detailed vulnerability data beyond just CVEs. Cloud Agents enable continuous assessment while scanner appliances handle network-based discovery.
