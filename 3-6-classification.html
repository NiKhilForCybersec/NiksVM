Asset Classification & Criticality

ğŸ“– Section 3.6 â±ï¸ 16 min read
ğŸ“˜ What You'll Learn

This page covers asset classification including criticality tiering models, business impact analysis, data sensitivity classification, exposure factors, and using classification for vulnerability prioritization.

Why Classification Matters

Not all assets are equal. A critical vulnerability on a production database server is far more urgent than the same vulnerability on a developer's test VM. Asset classification enables risk-based prioritization that aligns with business impact.

ğŸ’œ Classification = Prioritization

Without asset classification, you're treating all systems equally and prioritizing only by CVSS score. This leads to wasted effort patching low-value systems while critical infrastructure remains vulnerable.

Classification Automation

"""
Asset Criticality Classification Engine
Automatically classify assets based on configurable criteria.
"""

from typing import Dict, List
from dataclasses import dataclass
from enum import Enum


class CriticalityTier(Enum):
    TIER_1 = 1  # Critical
    TIER_2 = 2  # High
    TIER_3 = 3  # Medium
    TIER_4 = 4  # Low


@dataclass
class ClassificationResult:
    asset_id: str
    tier: CriticalityTier
    score: int
    factors: Dict[str, int]
    recommendations: List[str]


class AssetClassifier:
    """Classify assets into criticality tiers."""
    
    # Classification rules
    RULES = {
        'application_type': {
            'payment_processing': 40,
            'customer_facing': 35,
            'financial_system': 35,
            'identity_provider': 40,
            'database': 25,
            'web_application': 20,
            'internal_app': 10,
            'development': 5
        },
        'environment': {
            'production': 30,
            'staging': 15,
            'development': 5,
            'test': 5,
            'sandbox': 0
        },
        'data_classification': {
            'restricted': 40,
            'confidential': 30,
            'internal': 15,
            'public': 0
        },
        'network_zone': {
            'internet_facing': 30,
            'dmz': 25,
            'internal': 10,
            'isolated': 5
        },
        'compliance_scope': {
            'pci_dss': 15,
            'hipaa': 15,
            'sox': 10,
            'gdpr': 10
        }
    }
    
    def classify(self, asset: Dict) -> ClassificationResult:
        """
        Classify an asset based on its attributes.
        
        Args:
            asset: Asset dictionary with classification attributes
            
        Returns:
            ClassificationResult with tier and score breakdown
        """
        score = 0
        factors = {}
        recommendations = []
        
        # Score application type
        app_type = asset.get('application_type', '').lower()
        if app_type in self.RULES['application_type']:
            app_score = self.RULES['application_type'][app_type]
            score += app_score
            factors['application_type'] = app_score
        
        # Score environment
        env = asset.get('environment', '').lower()
        if env in self.RULES['environment']:
            env_score = self.RULES['environment'][env]
            score += env_score
            factors['environment'] = env_score
        
        # Score data classification
        data_class = asset.get('data_classification', '').lower()
        if data_class in self.RULES['data_classification']:
            data_score = self.RULES['data_classification'][data_class]
            score += data_score
            factors['data_classification'] = data_score
        else:
            recommendations.append('Assign data classification')
        
        # Score network exposure
        network = asset.get('network_zone', '').lower()
        if network in self.RULES['network_zone']:
            network_score = self.RULES['network_zone'][network]
            score += network_score
            factors['network_zone'] = network_score
        
        # Score compliance scope (can have multiple)
        compliance_score = 0
        for scope in asset.get('compliance_scope', []):
            scope_lower = scope.lower()
            if scope_lower in self.RULES['compliance_scope']:
                compliance_score += self.RULES['compliance_scope'][scope_lower]
        score += compliance_score
        factors['compliance_scope'] = compliance_score
        
        # Determine tier
        if score >= 80:
            tier = CriticalityTier.TIER_1
        elif score >= 50:
            tier = CriticalityTier.TIER_2
        elif score >= 25:
            tier = CriticalityTier.TIER_3
        else:
            tier = CriticalityTier.TIER_4
        
        # Validation recommendations
        if not asset.get('owner'):
            recommendations.append('Assign asset owner')
        if not asset.get('environment'):
            recommendations.append('Specify environment')
        
        return ClassificationResult(
            asset_id=asset.get('id', 'unknown'),
            tier=tier,
            score=score,
            factors=factors,
            recommendations=recommendations
        )
    
    def bulk_classify(self, assets: List[Dict]) -> Dict:
        """Classify multiple assets and return summary."""
        results = []
        tier_counts = {t: 0 for t in CriticalityTier}
        
        for asset in assets:
            result = self.classify(asset)
            results.append(result)
            tier_counts[result.tier] += 1
        
        return {
            'total_assets': len(assets),
            'tier_distribution': {t.name: count for t, count in tier_counts.items()},
            'results': results
        }


# Example usage
classifier = AssetClassifier()

assets = [
    {
        'id': 'srv-web-01',
        'application_type': 'customer_facing',
        'environment': 'production',
        'data_classification': 'confidential',
        'network_zone': 'dmz',
        'compliance_scope': ['pci_dss'],
        'owner': 'web-team'
    },
    {
        'id': 'srv-dev-01',
        'application_type': 'development',
        'environment': 'development',
        'data_classification': 'internal',
        'network_zone': 'internal'
    }
]

summary = classifier.bulk_classify(assets)
for result in summary['results']:
    print(f"{result.asset_id}: {result.tier.name} (Score: {result.score})")

Using Classification for Prioritization

SQL 
-- Risk-Based Prioritization Query
-- Combines vulnerability severity with asset criticality

WITH priority_vulns AS (
    SELECT 
        v.vulnerability_id,
        v.cve_id,
        v.cvss_score,
        v.severity,
        a.asset_id,
        a.hostname,
        a.criticality_tier,
        a.environment,
        a.network_zone,
        -- Calculate composite risk score
        CASE 
            WHEN v.severity = 'Critical' THEN 40
            WHEN v.severity = 'High' THEN 25
            WHEN v.severity = 'Medium' THEN 10
            WHEN v.severity = 'Low' THEN 3
            ELSE 1
        END AS severity_score,
        CASE 
            WHEN a.criticality_tier = 1 THEN 4
            WHEN a.criticality_tier = 2 THEN 3
            WHEN a.criticality_tier = 3 THEN 2
            ELSE 1
        END AS criticality_multiplier,
        CASE 
            WHEN a.network_zone = 'internet_facing' THEN 2
            WHEN a.network_zone = 'dmz' THEN 1.5
            ELSE 1
        END AS exposure_multiplier
    FROM vulnerabilities v
    JOIN assets a ON v.asset_id = a.asset_id
    WHERE v.status = 'Open'
)
SELECT 
    vulnerability_id,
    cve_id,
    hostname,
    severity,
    criticality_tier,
    -- Composite risk score
    ROUND(severity_score * criticality_multiplier * exposure_multiplier, 1) AS risk_score,
    -- Priority ranking
    RANK() OVER (ORDER BY 
        severity_score * criticality_multiplier * exposure_multiplier DESC
    ) AS priority_rank
FROM priority_vulns
ORDER BY risk_score DESC
LIMIT 100;

-- Get remediation workload by tier
SELECT 
    a.criticality_tier,
    v.severity,
    COUNT(*) as vuln_count,
    COUNT(DISTINCT a.asset_id) as affected_assets
FROM vulnerabilities v
JOIN assets a ON v.asset_id = a.asset_id
WHERE v.status = 'Open'
GROUP BY a.criticality_tier, v.severity
ORDER BY a.criticality_tier, 
    CASE v.severity 
        WHEN 'Critical' THEN 1 
        WHEN 'High' THEN 2 
        WHEN 'Medium' THEN 3 
        ELSE 4 
    END;
