1. PATCH / UPDATE
â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
   Apply vendor-provided security patch or version upgrade.
   
   Pros: Eliminates vulnerability, supported solution
   Cons: Requires testing, potential downtime, compatibility issues
   
   Best for: Most vulnerabilities with available patches

2. CONFIGURATION CHANGE
â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
   Modify settings to disable vulnerable feature or harden system.
   
   Pros: No software change, often immediate
   Cons: May reduce functionality, not always possible
   
   Examples:
   â€¢ Disable SMBv1, TLS 1.0/1.1
   â€¢ Remove unnecessary services
   â€¢ Restrict network access
   â€¢ Enable security features

3. WORKAROUND
â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
   Vendor-provided temporary mitigation (not full fix).
   
   Pros: Faster than patch, reduces immediate risk
   Cons: Temporary, may impact functionality
   
   Examples:
   â€¢ Microsoft's Log4j mitigation scripts
   â€¢ Registry modifications
   â€¢ Feature disablement

4. COMPENSATING CONTROL
â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
   Implement alternative security measure to reduce risk.
   
   Pros: Addresses risk when patching not possible
   Cons: Doesn't eliminate vulnerability, requires monitoring
   
   Examples:
   â€¢ WAF rules blocking exploit
   â€¢ Network segmentation
   â€¢ Enhanced monitoring/detection
   â€¢ IPS signatures

5. ACCEPT RISK
â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
   Formally accept residual risk (with documentation).
   
   Pros: Acknowledges reality when remediation not feasible
   Cons: Risk remains, requires governance
   
   Requirements:
   â€¢ Business owner approval
   â€¢ Documented justification
   â€¢ Compensating controls
   â€¢ Review timeline

"""
Remediation Planning Engine
Generate remediation plans based on vulnerability and asset context.
"""

from typing import Dict, List
from dataclasses import dataclass
from enum import Enum
from datetime import datetime, timedelta


class RemediationType(Enum):
    PATCH = "patch"
    CONFIG_CHANGE = "config_change"
    WORKAROUND = "workaround"
    COMPENSATING_CONTROL = "compensating_control"
    RISK_ACCEPTANCE = "risk_acceptance"


@dataclass
class RemediationPlan:
    vulnerability_id: str
    asset_id: str
    remediation_type: RemediationType
    description: str
    steps: List[str]
    estimated_effort: str
    requires_downtime: bool
    requires_change_approval: bool
    target_date: datetime
    owner: str
    fallback_plan: str


class RemediationPlanner:
    """Generate remediation plans for vulnerabilities."""
    
    # SLA by priority
    SLA_DAYS = {
        'P1': 2,
        'P2': 7,
        'P3': 30,
        'P4': 90,
        'P5': 180
    }
    
    def generate_plan(self, vuln: Dict, asset: Dict, 
                      priority: str) -> RemediationPlan:
        """
        Generate remediation plan based on context.
        
        Args:
            vuln: Vulnerability details
            asset: Asset details
            priority: Priority level (P1-P5)
        """
        # Determine remediation type
        rem_type = self._determine_type(vuln, asset)
        
        # Generate steps based on type
        steps = self._generate_steps(rem_type, vuln, asset)
        
        # Calculate target date
        sla_days = self.SLA_DAYS.get(priority, 30)
        target_date = datetime.now() + timedelta(days=sla_days)
        
        # Determine if downtime needed
        requires_downtime = self._requires_downtime(rem_type, asset)
        
        # Determine owner
        owner = asset.get('owner', 'IT Operations')
        
        return RemediationPlan(
            vulnerability_id=vuln.get('id'),
            asset_id=asset.get('id'),
            remediation_type=rem_type,
            description=self._generate_description(rem_type, vuln),
            steps=steps,
            estimated_effort=self._estimate_effort(rem_type, asset),
            requires_downtime=requires_downtime,
            requires_change_approval=priority in ['P1', 'P2'] or requires_downtime,
            target_date=target_date,
            owner=owner,
            fallback_plan=self._generate_fallback(rem_type)
        )
    
    def _determine_type(self, vuln: Dict, asset: Dict) -> RemediationType:
        """Determine best remediation approach."""
        # Check if patch available
        if vuln.get('patch_available', True):
            # Check if asset can be patched
            if not asset.get('patch_restricted', False):
                return RemediationType.PATCH
        
        # Check for workaround
        if vuln.get('workaround_available'):
            return RemediationType.WORKAROUND
        
        # Check if config change possible
        if vuln.get('config_mitigation'):
            return RemediationType.CONFIG_CHANGE
        
        # Default to compensating control
        return RemediationType.COMPENSATING_CONTROL
    
    def _generate_steps(self, rem_type: RemediationType, 
                        vuln: Dict, asset: Dict) -> List[str]:
        """Generate remediation steps."""
        if rem_type == RemediationType.PATCH:
            return [
                f"1. Obtain patch: {vuln.get('patch_id', 'See vendor advisory')}",
                "2. Review patch notes and known issues",
                "3. Test in non-production environment",
                "4. Schedule maintenance window",
                "5. Create system backup/snapshot",
                "6. Apply patch",
                "7. Verify system functionality",
                "8. Rescan to confirm remediation"
            ]
        
        elif rem_type == RemediationType.CONFIG_CHANGE:
            return [
                f"1. Document current configuration",
                f"2. Apply configuration: {vuln.get('config_mitigation', 'See advisory')}",
                "3. Test system functionality",
                "4. Update configuration management",
                "5. Rescan to confirm remediation"
            ]
        
        elif rem_type == RemediationType.WORKAROUND:
            return [
                f"1. Review vendor workaround documentation",
                "2. Assess impact on functionality",
                "3. Apply workaround",
                "4. Document temporary state",
                "5. Schedule full remediation when available",
                "6. Rescan to confirm risk reduction"
            ]
        
        elif rem_type == RemediationType.COMPENSATING_CONTROL:
            return [
                "1. Document why patching is not possible",
                "2. Identify compensating controls",
                "3. Implement network restrictions/segmentation",
                "4. Deploy monitoring/detection rules",
                "5. Request risk acceptance approval",
                "6. Set review date for reassessment"
            ]
        
        return ["1. Review vulnerability and determine approach"]
    
    def _requires_downtime(self, rem_type: RemediationType, 
                           asset: Dict) -> bool:
        """Determine if remediation requires downtime."""
        if rem_type == RemediationType.COMPENSATING_CONTROL:
            return False
        
        # Check asset type
        asset_type = asset.get('type', '').lower()
        if asset_type in ['server', 'database', 'network_device']:
            return True
        
        return False
    
    def _estimate_effort(self, rem_type: RemediationType, 
                         asset: Dict) -> str:
        """Estimate remediation effort."""
        estimates = {
            RemediationType.PATCH: "1-4 hours",
            RemediationType.CONFIG_CHANGE: "30 min - 2 hours",
            RemediationType.WORKAROUND: "1-2 hours",
            RemediationType.COMPENSATING_CONTROL: "4-8 hours",
            RemediationType.RISK_ACCEPTANCE: "2-4 hours (documentation)"
        }
        return estimates.get(rem_type, "Unknown")
    
    def _generate_description(self, rem_type: RemediationType, 
                              vuln: Dict) -> str:
        """Generate remediation description."""
        descriptions = {
            RemediationType.PATCH: f"Apply vendor patch for {vuln.get('cve_id', 'vulnerability')}",
            RemediationType.CONFIG_CHANGE: "Apply configuration mitigation",
            RemediationType.WORKAROUND: "Implement vendor-provided workaround",
            RemediationType.COMPENSATING_CONTROL: "Implement compensating controls",
            RemediationType.RISK_ACCEPTANCE: "Document and accept residual risk"
        }
        return descriptions.get(rem_type, "Remediate vulnerability")
    
    def _generate_fallback(self, rem_type: RemediationType) -> str:
        """Generate fallback plan."""
        if rem_type == RemediationType.PATCH:
            return "If patch fails: Restore from backup, implement compensating controls"
        return "If primary approach fails: Escalate to security team"


# Example usage
planner = RemediationPlanner()

vuln = {
    'id': 'vuln-001',
    'cve_id': 'CVE-2024-1234',
    'patch_available': True,
    'patch_id': 'KB5001234'
}

asset = {
    'id': 'srv-web-01',
    'type': 'server',
    'owner': 'Web Team',
    'patch_restricted': False
}

plan = planner.generate_plan(vuln, asset, 'P2')
print(f"Remediation: {plan.remediation_type.value}")
print(f"Target: {plan.target_date.date()}")
print(f"Steps:")
for step in plan.steps:
    print(f"  {step}")
